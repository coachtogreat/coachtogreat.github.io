Pengantar JavaScript Modern (ES6+)

1. Pengenalan ES6+

Fitur-fitur ES6+ 
// 1. Arrow Function
const tambah = (a, b) => a + b;

// 2. Template Literals
const nama = "John";
const pesan = `Halo, ${nama}!`;

// 3. Destructuring
const user = {
    nama: "John",
    umur: 30
};

const { nama, umur } = user;

// 4. Default Parameters
function sayHello(nama = "Guest") {
    return `Halo, ${nama}!`;
}

// 5. Rest Parameters
function sum(...numbers) {
    return numbers.reduce((a, b) => a + b, 0);
}

// 6. Spread Operator
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const arr3 = [...arr1, ...arr2];

// 7. Classes
class User {
    constructor(nama, umur) {
        this.nama = nama;
        this.umur = umur;
    }
    
    sapa() {
        return `Halo, nama saya ${this.nama}`;
    }
}

// 8. Modules
// file: user.js
export class User {
    constructor(nama) {
        this.nama = nama;
    }
}

// file: main.js
import { User } from './user.js';

// 9. Async/Await
async function fetchData() {
    try {
        const response = await fetch('https://api.example.com/data');
        const data = await response.json();
        return data;
    } catch (error) {
        throw error;
    }
}

3. Best Practices

  • Gunakan const dan let untuk variabel
  • Gunakan arrow function untuk callback
  • Gunakan template literals untuk string
  • Gunakan destructuring untuk objek
  • Gunakan async/await untuk async code
  • Gunakan modules untuk organisasi kode
  • Gunakan class untuk objek yang berhubungan
  • Gunakan private fields untuk encapsulation

// 1. Aplikasi Todo dengan ES6+
// file: todo.js
export class Todo {
    constructor() {
        this.todos = [];
    }
    
    tambah(tugas) {
        this.todos.push({
            id: Date.now(),
            tugas: tugas,
            selesai: false
        });
    }
    
    hapus(id) {
        this.todos = this.todos.filter(todo => todo.id !== id);
    }
    
    selesai(id) {
        const todo = this.todos.find(todo => todo.id === id);
        if (todo) {
            todo.selesai = true;
        }
    }
}

// file: main.js
import { Todo } from './todo.js';

const todoApp = new Todo();

function tambahTodo() {
    const input = document.querySelector('#todo-input');
    const tugas = input.value;
    
    if (tugas) {
        todoApp.tambah(tugas);
        input.value = '';
        renderTodo();
    }
}

function renderTodo() {
    const todoList = document.querySelector('#todo-list');
    todoList.innerHTML = todoApp.todos
        .map(todo => `
            
  • 
                
                ${todo.tugas}
                
            
    • 
        `)
        .join('');
}

// 3. Aplikasi User dengan ES6+
// file: user.js
export class User {
    #nama;
    #email;
    
    constructor(nama, email) {
        this.#nama = nama;
        this.#email = email;
    }
    
    get nama() {
        return this.#nama;
    }
    
    get email() {
        return this.#email;
    }
    
    static validateEmail(email) {
        const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        return regex.test(email);
    }
}

// file: main.js
import { User } from './user.js';

function buatUser(nama, email) {
    if (!User.validateEmail(email)) {
        throw new Error('Email tidak valid');
    }
    return new User(nama, email);
}

// 4. Aplikasi Calculator dengan ES6+
// file: calculator.js
export class Calculator {
    static tambah(a, b) {
        return a + b;
    }
    
    static kurang(a, b) {
        return a - b;
    }
    
    static kali(a, b) {
        return a * b;
    }
    
    static bagi(a, b) {
        if (b === 0) {
            throw new Error('Tidak bisa membagi dengan nol');
        }
        return a / b;
    }
}

// file: main.js
import { Calculator } from './calculator.js';

function hitung(operasi, a, b) {
    switch (operasi) {
        case 'tambah':
            return Calculator.tambah(a, b);
        case 'kurang':
            return Calculator.kurang(a, b);
        case 'kali':
            return Calculator.kali(a, b);
        case 'bagi':
            return Calculator.bagi(a, b);
        default:
            throw new Error('Operasi tidak dikenal');
    }
}

// 5. Aplikasi Form dengan ES6+
// file: form.js
export class Form {
    constructor(formElement) {
        this.form = formElement;
        this.setupEventListeners();
    }
    
    setupEventListeners() {
        this.form.addEventListener('submit', (e) => {
            e.preventDefault();
            this.handleSubmit();
        });
    }
    
    handleSubmit() {
        const data = {};
        const inputs = this.form.querySelectorAll('input');
        
        inputs.forEach(input => {
            data[input.name] = input.value;
        });
        
        this.onSubmit(data);
    }
    
    onSubmit(data) {
        console.log('Data:', data);
    }
}

// file: main.js
import { Form } from './form.js';
