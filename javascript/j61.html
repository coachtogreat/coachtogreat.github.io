Fungsi Dasar dalam JavaScript

1. Pengenalan Fungsi

Syntax Fungsi 
// Fungsi declaration
function namaFungsi() {
    // kode yang dijalankan
}

// Fungsi expression
const namaFungsi = function() {
    // kode yang dijalankan
};

// Arrow function
const namaFungsi = () => {
    // kode yang dijalankan
};

2. Contoh Fungsi Sederhana

Contoh Sederhana 
// Fungsi sederhana
function sapa() {
    console.log("Halo!");
}
sapa(); // output: "Halo!"

// Fungsi dengan parameter
function sapaNama(nama) {
    console.log("Halo, " + nama + "!");
}
sapaNama("John"); // output: "Halo, John!"

// Fungsi dengan return
function tambah(a, b) {
    return a + b;
}
let hasil = tambah(5, 3); // hasil = 8

// Fungsi dengan nilai default
function sapaNama(nama = "Teman") {
    return "Halo, " + nama;
}
console.log(sapaNama()); // "Halo, Teman"
console.log(sapaNama("John")); // "Halo, John"

3. Scope dalam Fungsi

Contoh Scope 
// Scope lokal
function hitung() {
    let a = 5;
    console.log(a); // 5
}

hitung(); // 5
console.log(a); // Error: a is not defined

// Scope global
let x = 10;

function tampilkan() {
    console.log(x); // 10
}

tampilkan(); // 10

// Scope nested
function luar() {
    let luar = "luar";
    
    function dalam() {
        let dalam = "dalam";
        console.log(luar); // "luar"
        console.log(dalam); // "dalam"
    }
    
    dalam();
}

luar();

4. Hoisting dalam Fungsi

Contoh Hoisting 
// Hoisting fungsi declaration
sapa(); // Berhasil: "Halo!"
function sapa() {
    console.log("Halo!");
}

// Tidak hoisting fungsi expression
sapa(); // Error: sapa is not a function
let sapa = function() {
    console.log("Halo!");
};

// Arrow function tidak hoisting
sapa(); // Error: sapa is not a function
const sapa = () => {
    console.log("Halo!");
};

5. Best Practices

  • Beri nama fungsi yang deskriptif
  • Hindari nama fungsi yang terlalu panjang
  • Pisahkan fungsi yang kompleks
  • Gunakan parameter dengan bijak
  • Tambahkan komentar untuk fungsi yang kompleks
  • Hindari side effects

6. Contoh Praktis

Contoh Kalkulator 
// Fungsi kalkulator sederhana
function tambah(a, b) {
    return a + b;
}

function kurang(a, b) {
    return a - b;
}

function kali(a, b) {
    return a * b;
}

function bagi(a, b) {
    if (b === 0) {
        return "Tidak bisa dibagi nol";
    }
    return a / b;
}

// Penggunaan
console.log(tambah(5, 3)); // 8
console.log(kurang(10, 5)); // 5
console.log(kali(4, 3)); // 12
console.log(bagi(10, 2)); // 5

// Fungsi validasi form
function validateForm(nama, email, umur) {
    if (!nama) {
        return "Nama tidak boleh kosong";
    }
    if (!email.includes("@")) {
        return "Email tidak valid";
    }
    if (umur < 18) {
        return "Umur minimal 18 tahun";
    }
    return true;
}

// Penggunaan
console.log(validateForm("John", "john@example.com", 25)); // true
console.log(validateForm("", "john@example.com", 25)); // "Nama tidak boleh kosong"

// Fungsi untuk menghitung rata-rata
function hitungRataRata(nilai) {
    if (nilai.length === 0) {
        return 0;
    }
    let total = 0;
    for (let i = 0; i < nilai.length; i++) {
        total += nilai[i];
    }
    return total / nilai.length;
}

// Penggunaan
console.log(hitungRataRata([80, 75, 90, 85])); // 82.5

7. Tips Penggunaan

  • Buat fungsi yang fokus pada satu tugas
  • Pisahkan logika yang kompleks ke dalam fungsi
  • Gunakan parameter dengan bijak
  • Tambahkan komentar untuk menjelaskan fungsi
  • Hindari fungsi yang terlalu panjang
  • Gunakan return untuk mengembalikan nilai
