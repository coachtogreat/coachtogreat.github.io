For Loop dalam JavaScript

1. Pengenalan For Loop

Syntax For Loop 
for (inisialisasi; kondisi; increment/decrement) {
    // kode yang dijalankan
}

2. Contoh For Loop Sederhana

Contoh Sederhana 
// Contoh hitung mundur
for (let i = 5; i > 0; i--) {
    console.log(i);
}

// Contoh hitung maju
for (let i = 0; i < 5; i++) {
    console.log(i);
}

// Contoh dengan step 2
for (let i = 0; i < 10; i += 2) {
    console.log(i);
}

3. For Loop dengan Array

Contoh Array 
// Contoh array sederhana
let angka = [1, 2, 3, 4, 5];
for (let i = 0; i < angka.length; i++) {
    console.log(angka[i]);
}

// Contoh array dengan operasi
let nilai = [80, 75, 90, 85];
let total = 0;
for (let i = 0; i < nilai.length; i++) {
    total += nilai[i];
}
console.log("Rata-rata: " + (total / nilai.length));

// Contoh array dengan kondisi
let buah = ["apel", "mangga", "jeruk", "pisang"];
for (let i = 0; i < buah.length; i++) {
    if (buah[i] === "jeruk") {
        console.log("Ditemukan jeruk!");
    }
}

4. For Loop dengan Object

Contoh Object 
// Contoh object sederhana
let orang = {
    nama: "John",
    umur: 25,
    kota: "Jakarta"
};

// Mengakses properti
for (let prop in orang) {
    console.log(prop + ": " + orang[prop]);
}

// Contoh array of object
let siswa = [
    { nama: "John", nilai: 85 },
    { nama: "Jane", nilai: 90 },
    { nama: "Mike", nilai: 75 }
];

for (let i = 0; i < siswa.length; i++) {
    console.log(siswa[i].nama + " mendapat nilai " + siswa[i].nilai);
}

5. For Loop dengan String

Contoh String 
// Contoh menghitung huruf
let kata = "Hello";
let jumlahHuruf = 0;
for (let i = 0; i < kata.length; i++) {
    if (kata[i] !== " ") {
        jumlahHuruf++;
    }
}
console.log("Jumlah huruf: " + jumlahHuruf);

// Contoh reverse string
let nama = "John";
let namaReverse = "";
for (let i = nama.length - 1; i >= 0; i--) {
    namaReverse += nama[i];
}
console.log(namaReverse); // "nhoJ"

6. Best Practices

  • Mulai dengan nilai awal yang jelas
  • Gunakan variabel yang deskriptif
  • Pisahkan kondisi kompleks ke dalam fungsi
  • Gunakan let untuk variabel loop
  • Hindari nested loop yang terlalu dalam
  • Gunakan break dan continue dengan hati-hati

7. Contoh Praktis

Contoh Kalkulator 
// Fungsi hitung faktorial
function faktorial(n) {
    let hasil = 1;
    for (let i = 1; i <= n; i++) {
        hasil *= i;
    }
    return hasil;
}

// Fungsi hitung rata-rata
function hitungRataRata(nilai) {
    let total = 0;
    for (let i = 0; i < nilai.length; i++) {
        total += nilai[i];
    }
    return total / nilai.length;
}

// Contoh penggunaan
console.log(faktorial(5)); // 120
console.log(hitungRataRata([80, 75, 90, 85])); // 82.5

// Contoh menghitung jumlah huruf vokal
function hitungVokal(kata) {
    let vokal = "aiueoAIUEO";
    let jumlah = 0;
    for (let i = 0; i < kata.length; i++) {
        if (vokal.includes(kata[i])) {
            jumlah++;
        }
    }
    return jumlah;
}

console.log(hitungVokal("Hello World")); // 3
