Break dan Continue dalam JavaScript

1. Pengenalan Break

Contoh Break 
// Break dalam for loop
for (let i = 0; i < 10; i++) {
    if (i === 5) {
        break; // keluar dari loop saat i = 5
    }
    console.log(i); // output: 0, 1, 2, 3, 4
}

// Break dalam while loop
let i = 0;
while (true) {
    if (i === 5) {
        break; // keluar dari loop saat i = 5
    }
    console.log(i);
    i++;
}

// Break dalam switch
let nilai = 85;
switch (nilai) {
    case 100:
        console.log("Sempurna!");
        break;
    case 90:
        console.log("Sangat Baik!");
        break;
    case 80:
        console.log("Baik!");
        break;
    default:
        console.log("Cukup");
}

2. Pengenalan Continue

Contoh Continue 
// Continue dalam for loop
for (let i = 0; i < 10; i++) {
    if (i % 2 === 0) {
        continue; // skip angka genap
    }
    console.log(i); // output: 1, 3, 5, 7, 9
}

// Continue dalam while loop
let i = 0;
while (i < 10) {
    i++;
    if (i % 3 === 0) {
        continue; // skip kelipatan 3
    }
    console.log(i); // output: 1, 2, 4, 5, 7, 8, 10
}

// Continue dengan label
outer: for (let i = 0; i < 3; i++) {
    for (let j = 0; j < 3; j++) {
        if (i === 1 && j === 1) {
            continue outer; // skip ke loop outer
        }
        console.log(i + ", " + j);
    }
}

3. Perbandingan Break dan Continue

Perbandingan 
// Break: keluar dari loop
for (let i = 0; i < 5; i++) {
    if (i === 3) {
        break; // keluar saat i = 3
    }
    console.log(i); // output: 0, 1, 2
}

// Continue: skip iterasi
for (let i = 0; i < 5; i++) {
    if (i === 3) {
        continue; // skip saat i = 3
    }
    console.log(i); // output: 0, 1, 2, 4

4. Best Practices

  • Gunakan break untuk keluar dari loop
  • Gunakan continue untuk skip iterasi
  • Gunakan label dengan hati-hati
  • Tambahkan komentar untuk menjelaskan penggunaan
  • Hindari nested break/continue yang terlalu dalam
  • Pisahkan kondisi kompleks ke dalam fungsi

5. Contoh Praktis

Contoh Validasi 
// Validasi form dengan break
function validateForm(nama, email, umur) {
    if (!nama) {
        console.log("Nama tidak boleh kosong");
        return false;
    }
    if (!email.includes("@")) {
        console.log("Email tidak valid");
        return false;
    }
    if (umur < 18) {
        console.log("Umur minimal 18 tahun");
        return false;
    }
    return true;
}

// Filter array dengan continue
function filterAngka(deret) {
    let hasil = [];
    for (let i = 0; i < deret.length; i++) {
        if (deret[i] % 2 === 0) {
            continue; // skip angka genap
        }
        if (deret[i] > 100) {
            break; // berhenti jika lebih dari 100
        }
        hasil.push(deret[i]);
    }
    return hasil;
}

// Contoh game tebak angka dengan break
let jawaban = Math.floor(Math.random() * 10) + 1;
let tebakan;
let skor = 100;

console.log("Tebak angka antara 1-10!");

while (skor > 0) {
    tebakan = prompt("Masukkan tebakan Anda:");
    if (tebakan == jawaban) {
        console.log("Benar! Skor Anda: " + skor);
        break;
    }
    skor -= 10;
    console.log("Salah, coba lagi! Skor: " + skor);
}

6. Tips Penggunaan

  • Gunakan break untuk menghentikan pencarian
  • Gunakan continue untuk skip kondisi tertentu
  • Tambahkan komentar untuk menjelaskan logika
  • Hindari penggunaan yang terlalu kompleks
  • Pisahkan logika yang rumit ke dalam fungsi
  • Gunakan label dengan hati-hati
