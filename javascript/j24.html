Tipe Data Dasar dalam JavaScript

1. String

Contoh String 
// Membuat string
let nama = "John";
let pesan = 'Halo, Dunia!';
let teks = `Saya sedang belajar ${nama}`;

// Manipulasi string
console.log(nama.length); // 4
console.log(nama.toUpperCase()); // JOHN
console.log(teks.includes("belajar")); // true

2. Number

Contoh Number 
// Angka biasa
let umur = 25;

// Angka desimal
let harga = 19.99;

// Operasi matematika
let total = umur + harga;
console.log(total); // 44.99

// Konversi string ke number
let angka = Number("42");
console.log(angka); // 42

3. Boolean

Contoh Boolean 
// Nilai boolean
let isStudent = true;
let isWorking = false;

// Operasi boolean
console.log(isStudent && isWorking); // false
console.log(isStudent || isWorking); // true
console.log(!isStudent); // false

4. Null

Contoh Null 
// Nilai null
let data = null;
console.log(typeof data); // "object" (catatan: ini adalah bug di JavaScript)

// Memeriksa null
if (data === null) {
    console.log("Data kosong");
}

5. Undefined

Contoh Undefined 
// Variabel yang belum diberi nilai
let nama;
console.log(nama); // undefined

// Properti yang tidak ada
const orang = {
    nama: "John"
};
console.log(orang.umur); // undefined

// Fungsi yang tidak mengembalikan nilai
function halo() {
    console.log("Halo!");
}
console.log(halo()); // undefined

6. Perbedaan Null dan Undefined

Perbedaan Null Undefined
Arti Nilai yang kosong secara sengaja Nilai yang belum didefinisikan
Penulisan let x = null; let x;
typeof "object" "undefined"
Penggunaan Reset nilai Nilai default

7. Best Practices

  • Gunakan tipe data yang sesuai
  • Hindari penggunaan null kecuali diperlukan
  • Gunakan isNaN() untuk memeriksa NaN
  • Gunakan typeof untuk memeriksa tipe data
  • Gunakan == untuk perbandingan nilai
  • Gunakan === untuk perbandingan nilai dan tipe

8. Contoh Praktis

Contoh Praktis 
// Fungsi untuk menghitung diskon
function hitungDiskon(harga, persen = 10) {
    if (harga === null || harga === undefined) {
        return "Harga tidak valid";
    }
    
    let diskon = harga * (persen / 100);
    return harga - diskon;
}

// Contoh penggunaan
console.log(hitungDiskon(100000)); // 90000
console.log(hitungDiskon(100000, 20)); // 80000
console.log(hitungDiskon(null)); // "Harga tidak valid"
