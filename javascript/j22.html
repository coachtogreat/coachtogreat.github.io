2. Komentar Multi Baris

Komentar Multi Baris 
/*
   Ini adalah komentar multi baris
   Cocok untuk penjelasan yang lebih panjang
   atau dokumentasi fungsi
*/

function hitungLuasPersegi(sisi) {
    /*
        Fungsi untuk menghitung luas persegi
        Parameter:
        - sisi: panjang sisi persegi
        
        Return:
        - hasil: luas persegi
    */
    let hasil = sisi * sisi;
    return hasil;
}

4. Komentar untuk Penjelasan Kode

Komentar Penjelasan Kode 
// Fungsi untuk menghitung diskon
function hitungDiskon(harga, persen) {
    // Hitung jumlah diskon
    let diskon = harga * (persen / 100);
    
    // Hitung harga setelah diskon
    let hargaAkhir = harga - diskon;
    
    // Kembalikan hasil
    return hargaAkhir;
}

5. Komentar untuk Dokumentasi

Komentar Dokumentasi 
/**
 * Fungsi untuk menghitung rata-rata array
 * 
 * @param {number[]} arr - Array yang berisi angka
 * @returns {number} - Rata-rata dari array
 * 
 * Contoh penggunaan:
 * ```
 * let hasil = rataRata([1, 2, 3, 4]);
 * console.log(hasil); // Output: 2.5
 * ```
 */
function rataRata(arr) {
    let total = 0;
    for (let i = 0; i < arr.length; i++) {
        total += arr[i];
    }
    return total / arr.length;
}

6. Best Practices untuk Komentar

  • Jelaskan "mengapa" bukan "bagaimana"
  • Komentari kode yang kompleks
  • Gunakan komentar untuk dokumentasi fungsi
  • Jangan over-comment kode sederhana
  • Update komentar saat kode diubah
  • Gunakan komentar untuk menonaktifkan kode sementara

7. Tips Penggunaan Komentar

  • Mulai dengan komentar yang menjelaskan tujuan program
  • Tambahkan komentar untuk setiap fungsi yang kompleks
  • Gunakan komentar untuk menjelaskan algoritma yang rumit
  • Komentari kode yang mungkin membingungkan
  • Gunakan komentar untuk menandai bagian kode yang perlu diperbaiki
