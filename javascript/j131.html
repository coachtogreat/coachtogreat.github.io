Try, Catch, Finally dalam JavaScript

1. Pengenalan Try, Catch, Finally

Contoh Dasar 
// Contoh try-catch
try {
    // Kode yang mungkin error
    const hasil = undefined.value;
} catch (error) {
    // Menangani error
    console.error("Terjadi error:", error);
}

// Contoh try-catch-finally
try {
    // Kode yang mungkin error
    const hasil = undefined.value;
} catch (error) {
    // Menangani error
    console.error("Terjadi error:", error);
} finally {
    // Kode yang selalu dijalankan
    console.log("Selesai");
}

// Contoh dengan multiple catch
try {
    // Kode yang mungkin error
    const hasil = undefined.value;
} catch (error) {
    if (error instanceof TypeError) {
        console.error("TypeError:", error);
    } else {
        console.error("Error lain:", error);
    }
} finally {
    console.log("Selesai");
}

2. Jenis Error

Jenis Error 
// 1. ReferenceError
try {
    console.log(undefinedVariable); // Variable tidak didefinisikan
} catch (error) {
    console.error(error.name); // "ReferenceError"
}

// 2. TypeError
try {
    const hasil = undefined.value; // Akses property undefined
} catch (error) {
    console.error(error.name); // "TypeError"
}

// 3. SyntaxError
try {
    eval("let x = 5; y = 10"); // Syntax error
} catch (error) {
    console.error(error.name); // "SyntaxError"
}

// 4. RangeError
try {
    const array = new Array(-1); // Ukuran array negatif
} catch (error) {
    console.error(error.name); // "RangeError"
}

// 5. URIError
try {
    decodeURIComponent("%" + "1"); // URI invalid
} catch (error) {
    console.error(error.name); // "URIError"
}

3. Best Practices

  • Gunakan try-catch untuk error handling
  • Tambahkan finally untuk operasi akhir
  • Gunakan instanceof untuk mengecek jenis error
  • Tambahkan komentar untuk menjelaskan error handling
  • Pisahkan logika error handling ke dalam fungsi
  • Beri nama variabel yang deskriptif
  • Tambahkan validasi input

4. Contoh Praktis

Contoh Penggunaan 
// Contoh todo list
function tambahTodo(teks) {
    try {
        if (typeof teks !== "string") {
            throw new TypeError("Todo harus berupa string");
        }
        
        const li = document.createElement("li");
        li.textContent = teks;
        document.getElementById("todo-list").appendChild(li);
        
        return li;
    } catch (error) {
        console.error(error);
        return null;
    }
}

// Contoh galeri foto
function tampilkanGambar(url) {
    try {
        const img = new Image();
        img.src = url;
        
        if (!img.complete) {
            throw new Error("Gagal memuat gambar");
        }
        
        document.body.appendChild(img);
        return img;
    } catch (error) {
        console.error(error);
        return null;
    }
}

// Contoh form validation
function validateForm() {
    try {
        const nama = document.getElementById("nama");
        const email = document.getElementById("email");
        
        if (nama.value.trim().length === 0) {
            throw new Error("Nama tidak boleh kosong");
        }
        
        if (!email.value.includes("@")) {
            throw new Error("Email tidak valid");
        }
        
        return "Form valid";
    } catch (error) {
        console.error(error);
        return null;
    }
}

// Contoh animasi
function animasi(element) {
    try {
        if (!element) {
            throw new Error("Element tidak ditemukan");
        }
        
        let posisi = 0;
        const animasi = setInterval(() => {
            posisi++;
            element.style.left = posisi + "px";
            
            if (posisi >= 400) {
                clearInterval(animasi);
            }
        }, 10);
        
        return "Animasi dimulai";
    } catch (error) {
        console.error(error);
        return null;
    }
}

// Contoh fetch data
async function fetchData(url) {
    try {
        const response = await fetch(url);
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        const data = await response.json();
        return data;
    } catch (error) {
        console.error(error);
        return null;
    }
}
