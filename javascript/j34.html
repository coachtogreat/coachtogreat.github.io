Operator Penugasan dalam JavaScript

1. Operator Penugasan Dasar (=)

Contoh Penugasan Dasar 
// Penugasan sederhana
let x = 5;
let y = "Hello";
let z = true;

// Penugasan multiple
let a = b = c = 10;
console.log(a, b, c); // 10 10 10

// Penugasan dengan tipe data berbeda
let angka = 42;
let teks = "" + angka; // "42"
let boolean = !!angka; // true

2. Operator Penugasan Aritmatika

Contoh Operator Aritmatika 
// Penjumlahan
let a = 5;
a += 3; // sama dengan a = a + 3
console.log(a); // 8

// Pengurangan
let b = 10;
b -= 4; // sama dengan b = b - 4
console.log(b); // 6

// Perkalian
let c = 2;
c *= 3; // sama dengan c = c * 3
console.log(c); // 6

// Pembagian
let d = 10;
d /= 2; // sama dengan d = d / 2
console.log(d); // 5

// Sisa bagi
let e = 10;
e %= 3; // sama dengan e = e % 3
console.log(e); // 1

3. Operator Penugasan Bitwise

Contoh Operator Bitwise 
// AND bitwise
let a = 5; // 0101
a &= 3;    // 0011
console.log(a); // 1 (0001)

// OR bitwise
let b = 5; // 0101
b |= 3;    // 0011
console.log(b); // 7 (0111)

// XOR bitwise
let c = 5; // 0101
let d = 3; // 0011
c ^= d;    // 0110
console.log(c); // 6

// Shift left
let e = 5; // 0101
let f = e << 1; // 1010
console.log(f); // 10

4. Operator Penugasan Ternary

Contoh Operator Ternary 
// Contoh sederhana
let umur = 25;
let status = umur >= 18 ? "Dewasa" : "Anak-anak";

// Dengan assignment
let nilai = 75;
let grade = nilai >= 70 ? "A" : nilai >= 60 ? "B" : "C";

// Dengan kondisi kompleks
let jumlah = 10;
let hasil = jumlah > 0 ? jumlah : 0;

5. Operator Penugasan Nullish Coalescing

Contoh Nullish Coalescing 
// Contoh OR biasa
let nama = "";
let defaultName = nama || "User";
console.log(defaultName); // "User" (karena string kosong dianggap false)

// Contoh Nullish Coalescing
let nama = "";
let defaultName = nama ?? "User";
console.log(defaultName); // "" (karena string kosong bukan null/undefined)

// Contoh dengan object
let data = null;
let nilai = data ?? "Default";
console.log(nilai); // "Default"

6. Best Practices

  • Gunakan operator penugasan yang sesuai dengan kebutuhan
  • Hindari penggunaan terlalu banyak operator dalam satu baris
  • Gunakan tanda kurung untuk mengklarifikasi prioritas
  • Gunakan operator penugasan bitwise dengan hati-hati
  • Gunakan operator ternary untuk kondisi yang sederhana
  • Gunakan nullish coalescing untuk default value yang lebih aman

7. Contoh Praktis

Contoh Kalkulator Sederhana 
// Fungsi penjumlahan
function tambah(a, b) {
    return a + b;
}

// Fungsi pengurangan
function kurang(a, b) {
    return a - b;
}

// Fungsi perkalian
function kali(a, b) {
    return a * b;
}

// Fungsi pembagian
function bagi(a, b) {
    if (b === 0) return "Tidak bisa membagi dengan nol";
    return a / b;
}

// Contoh penggunaan
let hasil = 0;
hasil += tambah(5, 3); // 8
hasil -= kurang(10, 4); // -2
hasil *= kali(2, 3); // -6
hasil /= bagi(10, 2); // -0.6
