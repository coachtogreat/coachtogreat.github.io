Hoisting dalam JavaScript

1. Pengenalan Hoisting

Contoh Hoisting 
// Contoh hoisting variabel
console.log(halo); // undefined
var halo = "Halo!";

// Contoh hoisting fungsi
sapa(); // "Halo!"
function sapa() {
    console.log("Halo!");
}

// Contoh hoisting fungsi expression
// Error: sapa is not a function
terima();
const terima = function() {
    console.log("Terima kasih!");
};

2. Hoisting Variabel

Contoh Hoisting Variabel 
// Hoisting dengan var
console.log(nilai); // undefined
var nilai = 10;

// Hoisting dengan let/const
// Error: Cannot access 'nilai' before initialization
console.log(nilai);
let nilai = 10;

// Contoh lebih lengkap
function contoh() {
    console.log(x); // undefined
    console.log(y); // ReferenceError
    console.log(z); // ReferenceError
    
    var x = 10;
    let y = 20;
    const z = 30;
}

contoh();

3. Hoisting Fungsi

Contoh Hoisting Fungsi 
// Hoisting function declaration
sapa(); // "Halo!"
function sapa() {
    console.log("Halo!");
}

// Tidak hoisted: function expression
// Error: sapa is not a function
sapa();
const sapa = function() {
    console.log("Halo!");
};

// Contoh lebih kompleks
function contoh() {
    sapa(); // "Halo!"
    
    function sapa() {
        console.log("Halo!");
    }
    
    // Error: sapa is not a function
    terima();
    
    const terima = function() {
        console.log("Terima kasih!");
    };
}

contoh();

4. Best Practices

  • Definisikan variabel dan fungsi di awal kode
  • Gunakan let/const untuk variabel baru
  • Hindari penggunaan var jika tidak diperlukan
  • Gunakan function declaration untuk fungsi yang penting
  • Tambahkan komentar untuk menjelaskan hoisting
  • Pisahkan logika yang rumit ke dalam fungsi

5. Contoh Praktis

Contoh Penggunaan 
// Contoh kalkulator dengan hoisting
function tambah(a, b) {
    return a + b;
}

function kurang(a, b) {
    return a - b;
}

// Tidak perlu khawatir karena function declaration hoisted
console.log(tambah(5, 3)); // 8
console.log(kurang(10, 4)); // 6

// Contoh game dengan hoisting
function game() {
    let skor = 0;
    let nyawa = 3;
    
    function tambahSkor() {
        skor += 10;
        return skor;
    }
    
    function kurangNyawa() {
        nyawa--;
        return nyawa;
    }
    
    return {
        tambahSkor: tambahSkor,
        kurangNyawa: kurangNyawa,
        getSkor: function() {
            return skor;
        },
        getNyawa: function() {
            return nyawa;
        }
    };
}

// Contoh event handler
function tambah() {
    let jumlah = parseInt(tombol.dataset.jumlah) || 0;
    tombol.dataset.jumlah = jumlah + 1;
    tombol.textContent = "Klik: " + jumlah;
}

// Tidak perlu khawatir karena function declaration hoisted
tombol.addEventListener("click", tambah);

  • Definisikan variabel dan fungsi di awal kode
  • Gunakan let/const untuk variabel baru
  • Hindari penggunaan var jika tidak diperlukan
  • Gunakan function declaration untuk fungsi yang penting
  • Tambahkan komentar untuk menjelaskan hoisting
  • Pisahkan logika yang rumit ke dalam fungsi
