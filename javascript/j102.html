Seleksi Elemen dalam JavaScript

1. getElementById

Contoh getElementById 
// HTML
<div id="konten">Konten</div>

// JavaScript
const div = document.getElementById("konten");
console.log(div); // Element div

// Mengubah konten
div.textContent = "Konten baru";

// Mengubah style
div.style.color = "red";

// Mengubah class
div.className = "baru";

// Menghapus element
div.remove();

2. getElementsByClassName

Contoh getElementsByClassName 
// HTML
<div class="item">Item 1</div>
<div class="item">Item 2</div>

// JavaScript
const items = document.getElementsByClassName("item");
console.log(items); // HTMLCollection

// Mengubah semua item
for (let i = 0; i < items.length; i++) {
    items[i].style.color = "blue";
}

// Mengubah item pertama
items[0].style.color = "red";

// Mengubah item terakhir
items[items.length - 1].style.color = "green";

3. querySelector

Contoh querySelector 
// HTML
<div class="konten">
    <p class="paragraf">Paragraf 1</p>
    <p class="paragraf">Paragraf 2</p>
</div>

// JavaScript
// Seleksi satu element
const paragraf1 = document.querySelector(".paragraf");
console.log(paragraf1); // Element paragraf pertama

// Seleksi dengan kombinasi
const paragrafDalamDiv = document.querySelector("div .paragraf");
console.log(paragrafDalamDiv); // Element paragraf pertama

// Seleksi dengan atribut
const link = document.querySelector("a[href='https://webiddesign.com']");
console.log(link); // Element link

4. querySelectorAll

Contoh querySelectorAll 
// HTML
<div class="konten">
    <p class="paragraf">Paragraf 1</p>
    <p class="paragraf">Paragraf 2</p>
</div>

// JavaScript
// Seleksi semua element
const paragraf = document.querySelectorAll(".paragraf");
console.log(paragraf); // NodeList

// Mengubah semua element
paragraf.forEach(p => {
    p.style.color = "blue";
});

// Mengubah element pertama
paragraf[0].style.color = "red";

// Mengubah element terakhir
paragraf[paragraf.length - 1].style.color = "green";

// Seleksi dengan kombinasi
const paragrafDalamDiv = document.querySelectorAll("div .paragraf");
console.log(paragrafDalamDiv); // NodeList

5. Best Practices

  • Gunakan querySelector untuk seleksi satu element
  • Gunakan querySelectorAll untuk seleksi multiple element
  • Beri nama variabel yang deskriptif
  • Tambahkan komentar untuk menjelaskan seleksi
  • Pisahkan logika yang rumit ke dalam fungsi
  • Gunakan class untuk seleksi multiple element

6. Contoh Praktis

Contoh Penggunaan 
// Contoh form validation
function validateForm() {
    const form = document.getElementById("form");
    const nama = form.querySelector("#nama");
    const email = form.querySelector("#email");
    
    if (nama.value.trim().length === 0) {
        const pesan = document.createElement("div");
        pesan.textContent = "Nama tidak boleh kosong";
        pesan.className = "error";
        nama.parentNode.insertBefore(pesan, nama);
    }
}

// Contoh galeri foto
function buatGaleri() {
    const konten = document.querySelector(".galeri");
    const gambar = [
        { src: "gambar1.jpg", alt: "Gambar 1" },
        { src: "gambar2.jpg", alt: "Gambar 2" }
    ];
    
    gambar.forEach(img => {
        const element = document.createElement("img");
        element.src = img.src;
        element.alt = img.alt;
        konten.appendChild(element);
    });
}

// Contoh todo list
function tambahTodo() {
    const input = document.querySelector("#todo-input");
    const list = document.querySelector("#todo-list");
    
    if (input.value.trim()) {
        const li = document.createElement("li");
        li.textContent = input.value;
        list.appendChild(li);
        input.value = "";
    }
}

// Contoh event handling
function tambahEvent() {
    const tombol = document.querySelectorAll(".tombol");
    tombol.forEach(btn => {
        btn.addEventListener("click", function() {
            this.style.backgroundColor = "blue";
        });
    });
}
