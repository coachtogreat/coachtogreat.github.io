Konsep Modularisasi dalam JavaScript

1. Pengenalan Modularisasi

Contoh Dasar 
// 1. Modularisasi dengan Object Literal
const todo = {
    tambah: function(teks) {
        // Logika tambah todo
    },
    hapus: function(id) {
        // Logika hapus todo
    },
    tampilkan: function() {
        // Logika tampilkan todo
    }
};

// 2. Modularisasi dengan Constructor Function
function Todo() {
    const todos = [];
    
    this.tambah = function(teks) {
        todos.push(teks);
    };
    
    this.hapus = function(id) {
        todos.splice(id, 1);
    };
    
    this.tampilkan = function() {
        return todos;
    };
}

// 3. Modularisasi dengan IIFE
const todo = (function() {
    const todos = [];
    
    return {
        tambah: function(teks) {
            todos.push(teks);
        },
        hapus: function(id) {
            todos.splice(id, 1);
        },
        tampilkan: function() {
            return todos;
        }
    };
})();

// 4. Modularisasi dengan Class
class Todo {
    constructor() {
        this.todos = [];
    }
    
    tambah(teks) {
        this.todos.push(teks);
    }
    
    hapus(id) {
        this.todos.splice(id, 1);
    }
    
    tampilkan() {
        return this.todos;
    }
}

3. Pola Modularisasi

Pola Modularisasi 
// 1. Module Pattern
const todo = (function() {
    const todos = [];
    
    function tambahTodo(teks) {
        todos.push(teks);
    }
    
    function hapusTodo(id) {
        todos.splice(id, 1);
    }
    
    return {
        tambah: tambahTodo,
        hapus: hapusTodo,
        tampilkan: function() {
            return todos;
        }
    };
})();

// 2. Revealing Module Pattern
const todo = (function() {
    const todos = [];
    
    function tambahTodo(teks) {
        todos.push(teks);
    }
    
    function hapusTodo(id) {
        todos.splice(id, 1);
    }
    
    function tampilkanTodos() {
        return todos;
    }
    
    return {
        tambah: tambahTodo,
        hapus: hapusTodo,
        tampilkan: tampilkanTodos
    };
})();

// 3. Singleton Pattern

4. Best Practices

  • Pisahkan logika ke dalam module yang terpisah
  • Beri nama module yang deskriptif
  • Tambahkan komentar untuk menjelaskan module
  • Hindari global scope pollution
  • Gunakan pola yang sesuai dengan kebutuhan
  • Tambahkan error handling
  • Buat module yang mudah direuse

5. Contoh Praktis

Contoh Penggunaan 
// 1. Module untuk Todo List
const todo = (function() {
    const todos = [];
    
    function tambahTodo(teks) {
        if (typeof teks !== "string") {
            throw new Error("Todo harus berupa string");
        }
        todos.push(teks);
    }
    
    function hapusTodo(id) {
        if (typeof id !== "number") {
            throw new Error("ID harus berupa number");
        }
        todos.splice(id, 1);
    }
    
    return {
        tambah: tambahTodo,
        hapus: hapusTodo,
        tampilkan: function() {
            return todos;
        }
    };
})();

// 2. Module untuk Galeri Foto
const galeri = (function() {
    const gambar = [];
    
    function tambahGambar(url) {
        if (!url) {
            throw new Error("URL tidak boleh kosong");
        }
        gambar.push(url);
    }
    
    function hapusGambar(id) {
        if (typeof id !== "number") {
            throw new Error("ID harus berupa number");
        }
        gambar.splice(id, 1);
    }
    
    return {
        tambah: tambahGambar,
        hapus: hapusGambar,
        tampilkan: function() {
            return gambar;
        }
    };
})();

// 3. Module untuk Form Validation
const validator = (function() {
    function validateEmail(email) {
        const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        return regex.test(email);
    }
    
    function validatePassword(password) {
        return password.length >= 8;
    }
    
    return {
        email: validateEmail,
        password: validatePassword
    };
})();

// 4. Module untuk Animasi
const animasi = (function() {
    function gerakKiri(element, jarak) {
        element.style.left = jarak + "px";
    }
    
    function gerakKanan(element, jarak) {
        element.style.right = jarak + "px";
    }
    
    return {
        kiri: gerakKiri,
        kanan: gerakKanan
    };
})();

// 5. Module untuk Data Storage
const storage = (function() {
    function simpan(key, value) {
        localStorage.setItem(key, JSON.stringify(value));
    }
    
    function ambil(key) {
        const data = localStorage.getItem(key);
        return data ? JSON.parse(data) : null;
    }
    
    return {
        simpan: simpan,
        ambil: ambil
    };
})();
