Membuat Objek dalam JavaScript

1. Pengenalan Objek

Definisi Objek 
// Objek adalah struktur data yang menyimpan kumpulan properti
// Properti bisa berupa nilai atau fungsi

// Contoh objek sederhana
let orang = {
    nama: "John",
    umur: 25,
    alamat: "Jakarta"
};

// Objek dengan fungsi
let kalkulator = {
    tambah: function(a, b) {
        return a + b;
    },
    kurang: function(a, b) {
        return a - b;
    }
};

2. Cara Membuat Objek

Cara Membuat Objek 
// Menggunakan object literal
let orang = {
    nama: "John",
    umur: 25,
    alamat: "Jakarta"
};

// Menggunakan constructor Object
let orang = new Object();
orang.nama = "John";
orang.umur = 25;
orang.alamat = "Jakarta";

// Menggunakan function constructor
function Orang(nama, umur, alamat) {
    this.nama = nama;
    this.umur = umur;
    this.alamat = alamat;
}
let john = new Orang("John", 25, "Jakarta");

// Menggunakan class (ES6+)
class Orang {
    constructor(nama, umur, alamat) {
        this.nama = nama;
        this.umur = umur;
        this.alamat = alamat;
    }
}
let john = new Orang("John", 25, "Jakarta");

3. Mengakses Nilai Objek

Mengakses Nilai 
// Menggunakan dot notation
let orang = {
    nama: "John",
    umur: 25
};
console.log(orang.nama); // "John"
console.log(orang.umur); // 25

// Menggunakan bracket notation
console.log(orang["nama"]); // "John"
console.log(orang["umur"]); // 25

// Mengubah nilai
orang.nama = "Jane";
console.log(orang.nama); // "Jane"

// Menambah properti baru
orang.alamat = "Jakarta";
console.log(orang.alamat); // "Jakarta"

4. Objek dalam Objek

Objek dalam Objek 
// Objek sederhana
let siswa = {
    nama: "John",
    nilai: {
        matematika: 85,
        bahasa: 90,
        ipa: 88
    },
    alamat: {
        jalan: "Jl. Raya",
        kota: "Jakarta",
        kodePos: 12345
    }
};

// Mengakses nilai
console.log(siswa.nilai.matematika); // 85
console.log(siswa.alamat.kota); // "Jakarta"

// Mengubah nilai
siswa.nilai.bahasa = 95;
console.log(siswa.nilai.bahasa); // 95

// Menambah properti
siswa.nilai.ips = 80;
console.log(siswa.nilai.ips); // 80

5. Best Practices

  • Gunakan object literal untuk objek sederhana
  • Gunakan constructor untuk objek yang sering dibuat
  • Gunakan class untuk objek yang kompleks
  • Beri nama properti yang deskriptif
  • Tambahkan komentar untuk menjelaskan objek
  • Pisahkan objek yang kompleks ke dalam fungsi

6. Contoh Praktis

Contoh Penggunaan 
// Contoh data siswa
let siswa = {
    nama: "John",
    nilai: {
        matematika: 85,
        bahasa: 90,
        ipa: 88
    },
    alamat: {
        jalan: "Jl. Raya",
        kota: "Jakarta",
        kodePos: 12345
    },
    hitungRataRata: function() {
        let total = 0;
        let jumlah = 0;
        for (let nilai in this.nilai) {
            total += this.nilai[nilai];
            jumlah++;
        }
        return total / jumlah;
    }
};

// Contoh data produk
let produk = {
    id: 1,
    nama: "Laptop",
    harga: 10000000,
    stok: 10,
    diskon: 0.1,
    hitungHargaAkhir: function() {
        return this.harga * (1 - this.diskon);
    }
};

// Contoh data menu
let menu = {
    makanan: {
        "Nasi Goreng": {
            harga: 20000,
            stok: 50
        },
        "Mie Goreng": {
            harga: 25000,
            stok: 40
        }
    },
    minuman: {
        "Teh": {
            harga: 5000,
            stok: 100
        },
        "Kopi": {
            harga: 10000,
            stok: 80
        }
    }
};

// Contoh data transaksi
let transaksi = {
    tanggal: "2023-05-01",
    produk: [
        { nama: "Laptop", jumlah: 1 },
        { nama: "Smartphone", jumlah: 2 }
    ],
    hitungTotal: function() {
        let total = 0;
        for (let item of this.produk) {
            total += item.jumlah * this.harga(item.nama);
        }
        return total;
    }
};

7. Tips Penggunaan

  • Gunakan objek untuk menyimpan data terkait
  • Hindari objek yang terlalu dalam
  • Gunakan fungsi untuk operasi kompleks
  • Tambahkan komentar untuk menjelaskan objek
  • Pisahkan logika yang rumit ke dalam fungsi
  • Gunakan constructor atau class untuk objek yang sering dibuat
