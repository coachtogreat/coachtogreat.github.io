Operator Aritmatika dalam JavaScript

1. Operator Dasar

Operator Arti Contoh Hasil
+ Penjumlahan 5 + 3 8
- Pengurangan 10 - 4 6
* Perkalian 2 * 3 6
/ Pembagian 10 / 2 5
% Sisa bagi 10 % 3 1

2. Operator Increment dan Decrement

Contoh Increment/Decrement 
// Increment
let a = 5;
a++; // a = 6
a = a + 1; // a = 7

// Decrement
let b = 10;
b--; // b = 9
b = b - 1; // b = 8

// Pre vs Post increment
let x = 5;
console.log(x++); // output: 5 (post increment)
console.log(x); // output: 6

let y = 5;
console.log(++y); // output: 6 (pre increment)
console.log(y); // output: 6

3. Operator Pembagian

Contoh Pembagian 
// Pembagian biasa
let hasil = 10 / 3; // hasil: 3.3333333333333335

// Pembagian bulat
let bulat = Math.floor(10 / 3); // hasil: 3

// Pembagian bulat ke atas
let atas = Math.ceil(10 / 3); // hasil: 4

// Pembagian bulat ke bawah
let bawah = Math.trunc(10 / 3); // hasil: 3

4. Operator Sisa Bagi (%)

Contoh Sisa Bagi 
// Mengecek bilangan genap
function isEven(number) {
    return number % 2 === 0;
}

// Mengecek bilangan ganjil
function isOdd(number) {
    return number % 2 !== 0;
}

// Contoh penggunaan
console.log(isEven(4)); // true
console.log(isOdd(5)); // true

5. Operator Matematika Lanjutan

Contoh Operator Lanjutan 
// Pangkat (ES6+)
let kuadrat = 2 ** 3; // hasil: 8

// Akar kuadrat
let akar = Math.sqrt(16); // hasil: 4

// Nilai absolut
let abs = Math.abs(-5); // hasil: 5

// Pembulatan
let pembulatan = Math.round(3.7); // hasil: 4

6. Best Practices

  • Gunakan operator yang sesuai untuk situasi
  • Hati-hati dengan pembagian oleh nol
  • Gunakan Math.floor() untuk pembulatan ke bawah
  • Gunakan Math.ceil() untuk pembulatan ke atas
  • Gunakan operator % untuk menghitung sisa bagi
  • Gunakan operator ** untuk perpangkatan

7. Contoh Praktis

Contoh Kalkulator Sederhana 
// Fungsi penjumlahan
function tambah(a, b) {
    return a + b;
}

// Fungsi pengurangan
function kurang(a, b) {
    return a - b;
}

// Fungsi perkalian
function kali(a, b) {
    return a * b;
}

// Fungsi pembagian
function bagi(a, b) {
    if (b === 0) {
        return "Tidak bisa membagi dengan nol";
    }
    return a / b;
}

// Contoh penggunaan
console.log(tambah(5, 3)); // 8
console.log(kurang(10, 4)); // 6
console.log(kali(2, 3)); // 6
console.log(bagi(10, 2)); // 5
