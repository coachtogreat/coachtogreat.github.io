Function Expression vs Declaration dalam JavaScript

1. Function Declaration

Contoh Function Declaration 
// Sintaks Function Declaration
function namaFungsi() {
    // kode yang dijalankan
}

// Contoh sederhana
function sapa() {
    console.log("Halo!");
}

// Contoh dengan parameter
function tambah(a, b) {
    return a + b;
}

// Contoh dengan nilai default
function sapaNama(nama = "Teman") {
    return "Halo, " + nama;
}

// Hoisting pada Function Declaration
sapa(); // Berhasil: "Halo!"
function sapa() {
    console.log("Halo!");
}

2. Function Expression

Contoh Function Expression 
// Sintaks Function Expression
const namaFungsi = function() {
    // kode yang dijalankan
};

// Contoh sederhana
const sapa = function() {
    console.log("Halo!");
};

// Contoh dengan parameter
const tambah = function(a, b) {
    return a + b;
};

// Contoh dengan nilai default
const sapaNama = function(nama = "Teman") {
    return "Halo, " + nama;
};

// IIFE (Immediately Invoked Function Expression)
(function() {
    console.log("Ini adalah IIFE!");
})();

// Hoisting pada Function Expression
sapa(); // Error: sapa is not a function
const sapa = function() {
    console.log("Halo!");
};

3. Perbandingan Function Declaration vs Expression

Contoh Perbandingan 
// Function Declaration - bisa dipanggil sebelum didefinisikan
sapa(); // Berhasil: "Halo!"
function sapa() {
    console.log("Halo!");
}

// Function Expression - tidak bisa dipanggil sebelum didefinisikan
sapa(); // Error: sapa is not a function
const sapa = function() {
    console.log("Halo!");
};

// Function Expression sebagai parameter
function eksekusi(fungsi) {
    fungsi();
}

// Menggunakan Function Declaration
function sapa() {
    console.log("Halo!");
}
eksekusi(sapa);

// Menggunakan Function Expression
eksekusi(function() {
    console.log("Halo!");
});

4. Arrow Function

Contoh Arrow Function 
// Arrow Function sederhana
const sapa = () => {
    console.log("Halo!");
};

// Arrow Function dengan parameter
const tambah = (a, b) => {
    return a + b;
};

// Arrow Function dengan satu parameter
const kuadrat = x => x * x;

// Arrow Function dengan return implicit
const kali = (a, b) => a * b;

// Arrow Function dengan multiple statement
const hitung = (a, b) => {
    const total = a + b;
    return total;
};

// Arrow Function sebagai parameter
const angka = [1, 2, 3, 4, 5];
angka.map(x => x * 2); // [2, 4, 6, 8, 10]

5. Best Practices

  • Gunakan Function Declaration untuk fungsi yang sering dipanggil
  • Gunakan Function Expression untuk fungsi yang lebih kompleks
  • Gunakan Arrow Function untuk fungsi pendek dan sederhana
  • Hindari penggunaan yang terlalu kompleks
  • Tambahkan komentar untuk menjelaskan fungsi
  • Pisahkan logika yang rumit ke dalam fungsi

6. Contoh Praktis

Contoh Penggunaan 
// Contoh kalkulator dengan Function Declaration
function tambah(a, b) {
    return a + b;
}

function kurang(a, b) {
    return a - b;
}

function kali(a, b) {
    return a * b;
}

function bagi(a, b) {
    if (b === 0) {
        return "Tidak bisa dibagi nol";
    }
    return a / b;
}

// Contoh penggunaan Function Expression dalam objek
const kalkulator = {
    tambah: function(a, b) {
        return a + b;
    },
    kurang: function(a, b) {
        return a - b;
    },
    kali: function(a, b) {
        return a * b;
    },
    bagi: function(a, b) {
        if (b === 0) {
            return "Tidak bisa dibagi nol";
        }
        return a / b;
    }
};

// Contoh penggunaan Arrow Function dalam array
const angka = [1, 2, 3, 4, 5];

// Menggunakan Function Expression
angka.map(function(x) {
    return x * 2;
});

// Menggunakan Arrow Function
angka.map(x => x * 2);

// Contoh penggunaan dalam event handler
// Menggunakan Function Expression
button.addEventListener('click', function() {
    console.log("Button diklik!");
});

// Menggunakan Arrow Function
button.addEventListener('click', () => {
    console.log("Button diklik!");
});

// Contoh penggunaan dalam closure
function luar() {
    const x = 5;
    return function() {
        return x;
    };
}

const dalam = luar();
console.log(dalam()); // 5

7. Tips Penggunaan

  • Gunakan Function Declaration untuk fungsi yang sering dipanggil
  • Gunakan Function Expression untuk fungsi yang lebih kompleks
  • Gunakan Arrow Function untuk fungsi pendek dan sederhana
  • Hindari penggunaan yang terlalu kompleks
  • Tambahkan komentar untuk menjelaskan fungsi
  • Pisahkan logika yang rumit ke dalam fungsi
