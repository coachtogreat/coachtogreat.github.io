Metode Array Dasar dalam JavaScript

1. Metode push()

Contoh push() 
// Menambahkan elemen ke akhir array
let buah = ["apel", "mangga"];

// Menambah satu elemen
buah.push("jeruk");
console.log(buah); // ["apel", "mangga", "jeruk"]

// Menambah multiple elemen
buah.push("pisang", "semangka");
console.log(buah); // ["apel", "mangga", "jeruk", "pisang", "semangka"]

// Mengembalikan panjang array baru
let jumlah = buah.push("melon");
console.log(jumlah); // 6

2. Metode pop()

Contoh pop() 
// Menghapus elemen dari akhir array
let buah = ["apel", "mangga", "jeruk"];

// Menghapus satu elemen
let terakhir = buah.pop();
console.log(terakhir); // "jeruk"
console.log(buah); // ["apel", "mangga"]

// Menghapus lagi
let terakhir2 = buah.pop();
console.log(terakhir2); // "mangga"
console.log(buah); // ["apel"]

// Pop dari array kosong
let kosong = [];
let hasil = kosong.pop();
console.log(hasil); // undefined

3. Metode shift()

Contoh shift() 
// Menghapus elemen dari awal array
let buah = ["apel", "mangga", "jeruk"];

// Menghapus satu elemen
let pertama = buah.shift();
console.log(pertama); // "apel"
console.log(buah); // ["mangga", "jeruk"]

// Menghapus lagi
let pertama2 = buah.shift();
console.log(pertama2); // "mangga"
console.log(buah); // ["jeruk"]

// Shift dari array kosong
let kosong = [];
let hasil = kosong.shift();
console.log(hasil); // undefined

4. Metode unshift()

Contoh unshift() 
// Menambah elemen ke awal array
let buah = ["mangga", "jeruk"];

// Menambah satu elemen
buah.unshift("apel");
console.log(buah); // ["apel", "mangga", "jeruk"]

// Menambah multiple elemen
buah.unshift("pisang", "semangka");
console.log(buah); // ["pisang", "semangka", "apel", "mangga", "jeruk"]

// Mengembalikan panjang array baru
let jumlah = buah.unshift("melon");
console.log(jumlah); // 6

5. Contoh Penggunaan

Contoh Praktis 
// Contoh antrian (Queue)
let antrian = [];

// Menambah orang ke antrian
antrian.push("John");
console.log(antrian); // ["John"]

antrian.push("Jane");
console.log(antrian); // ["John", "Jane"]

// Mengambil orang dari antrian
let orang = antrian.shift();
console.log(orang); // "John"
console.log(antrian); // ["Jane"]

// Contoh tumpukan (Stack)
let tumpukan = [];

// Menambah ke tumpukan
console.log(tumpukan.push("apel")); // 1
console.log(tumpukan.push("mangga")); // 2

// Mengambil dari tumpukan
let buah = tumpukan.pop();
console.log(buah); // "mangga"
console.log(tumpukan); // ["apel"]

// Contoh history browser
let history = [];

// Menambah halaman
history.push("/home");
history.push("/about");

// Kembali ke halaman sebelumnya
let halaman = history.pop();
console.log(halaman); // "/about"

// Contoh chat
let pesan = [];

// Menambah pesan
pesan.push("Halo!");
pesan.push("Apa kabar?");

// Menghapus pesan terakhir
let terakhir = pesan.pop();
console.log(terakhir); // "Apa kabar?"

6. Best Practices

  • Gunakan push() untuk menambah elemen ke akhir array
  • Gunakan pop() untuk menghapus elemen dari akhir array
  • Gunakan shift() untuk menghapus elemen dari awal array
  • Gunakan unshift() untuk menambah elemen ke awal array
  • Tambahkan komentar untuk menjelaskan operasi
  • Pisahkan logika yang rumit ke dalam fungsi

7. Tips Penggunaan

  • Gunakan push() dan pop() untuk implementasi stack
  • Gunakan shift() dan unshift() untuk implementasi queue
  • Hati-hati dengan performa shift() dan unshift() pada array besar
  • Gunakan array method dengan bijak
  • Tambahkan komentar untuk menjelaskan operasi
  • Pisahkan logika yang rumit ke dalam fungsi
