Iterasi Array dalam JavaScript

1. Iterasi dengan For Loop

Contoh For Loop 
// Iterasi dengan for
let buah = ["apel", "mangga", "jeruk"];
for (let i = 0; i < buah.length; i++) {
    console.log(buah[i]);
}

// Iterasi dengan for...of
let angka = [1, 2, 3, 4, 5];
for (let nilai of angka) {
    console.log(nilai);
}

// Iterasi dengan for...in
let objek = { nama: "John", umur: 25 };
for (let prop in objek) {
    console.log(prop + ": " + objek[prop]);
}

2. Iterasi dengan forEach

Contoh forEach 
// forEach sederhana
let buah = ["apel", "mangga", "jeruk"];
buah.forEach(function(item) {
    console.log(item);
});

// forEach dengan index
buah.forEach(function(item, index) {
    console.log(index + ": " + item);
});

// forEach dengan array
let nilai = [80, 75, 90, 85];
nilai.forEach(function(nilai, index, array) {
    console.log("Nilai ke-" + index + ": " + nilai);
});

// forEach dengan arrow function
nilai.forEach((nilai, index) => {
    console.log("Nilai ke-" + index + ": " + nilai);
});

3. Perbandingan For vs forEach

Perbandingan 
// For lebih fleksibel
let buah = ["apel", "mangga", "jeruk"];
for (let i = 0; i < buah.length; i++) {
    if (buah[i] === "mangga") {
        break; // bisa menggunakan break
    }
    console.log(buah[i]);
}

// forEach lebih ringkas
buah.forEach(function(item) {
    console.log(item);
});

// For bisa mengubah array
for (let i = 0; i < buah.length; i++) {
    buah[i] = buah[i].toUpperCase();
}

// forEach tidak bisa mengubah array langsung
buah.forEach(function(item, index, array) {
    array[index] = item.toUpperCase();
});

4. Contoh Praktis

Contoh Penggunaan 
// Contoh kalkulator
let angka = [1, 2, 3, 4, 5];
let hasil = 0;

// Menggunakan for
for (let i = 0; i < angka.length; i++) {
    hasil += angka[i];
}
console.log(hasil); // 15

// Menggunakan forEach
hasil = 0;
angka.forEach(function(nilai) {
    hasil += nilai;
});
console.log(hasil); // 15

// Contoh validasi form
let input = ["John", "john@example.com", "25"];
let valid = true;

// Menggunakan for
for (let i = 0; i < input.length; i++) {
    if (input[i] === "") {
        valid = false;
        break;
    }
}

// Menggunakan forEach
valid = true;
input.forEach(function(item) {
    if (item === "") {
        valid = false;
    }
});

// Contoh pengolahan data
let siswa = [
    { nama: "John", nilai: 85 },
    { nama: "Jane", nilai: 90 },
    { nama: "Mike", nilai: 80 }
];

// Menggunakan for
let total = 0;
for (let i = 0; i < siswa.length; i++) {
    total += siswa[i].nilai;
}
console.log("Rata-rata: " + (total / siswa.length));

// Menggunakan forEach
let total2 = 0;
siswa.forEach(function(s) {
    total2 += s.nilai;
});
console.log("Rata-rata: " + (total2 / siswa.length));

// Contoh filter data
let angka = [1, 2, 3, 4, 5];
let genap = [];

// Menggunakan for
for (let i = 0; i < angka.length; i++) {
    if (angka[i] % 2 === 0) {
        genap.push(angka[i]);
    }
}

// Menggunakan forEach
let genap2 = [];
angka.forEach(function(n) {
    if (n % 2 === 0) {
        genap2.push(n);
    }
});

5. Best Practices

  • Gunakan forEach untuk operasi sederhana
  • Gunakan for untuk operasi yang lebih kompleks
  • Gunakan arrow function untuk forEach
  • Tambahkan komentar untuk menjelaskan operasi
  • Hindari side effects dalam forEach
  • Pisahkan logika yang rumit ke dalam fungsi
