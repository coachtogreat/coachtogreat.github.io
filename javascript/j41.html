If, Else, dan Else If dalam JavaScript

1. Statement If

Contoh Statement If 
// Contoh sederhana
let umur = 25;
if (umur >= 18) {
    console.log("Anda sudah dewasa");
}

// Dengan perbandingan
let nilai = 85;
if (nilai >= 70) {
    console.log("Nilai bagus!");
}

// Dengan operator logika
let login = true;
if (login && nilai >= 70) {
    console.log("Anda bisa mengakses sistem");
}

2. Statement If-Else

Contoh If-Else 
// Contoh sederhana
let umur = 16;
if (umur >= 18) {
    console.log("Anda sudah dewasa");
} else {
    console.log("Anda masih anak-anak");
}

// Dengan nilai boolean
let isLogin = false;
if (isLogin) {
    console.log("Selamat datang!");
} else {
    console.log("Silakan login terlebih dahulu");
}

// Dengan string
let nama = "";
if (nama) {
    console.log("Nama sudah diisi");
} else {
    console.log("Nama belum diisi");
}

3. Statement If-Else If-Else

Contoh If-Else If-Else 
// Contoh grade
let nilai = 85;
if (nilai >= 90) {
    console.log("Grade A");
} else if (nilai >= 80) {
    console.log("Grade B");
} else if (nilai >= 70) {
    console.log("Grade C");
} else {
    console.log("Grade D");
}

// Contoh umur
let umur = 25;
if (umur < 18) {
    console.log("Anak-anak");
} else if (umur >= 18 && umur < 30) {
    console.log("Dewasa muda");
} else if (umur >= 30 && umur < 60) {
    console.log("Dewasa");
} else {
    console.log("Lansia");
}

4. Nested If

Contoh Nested If 
// Contoh login dan role
let isLogin = true;
let role = "admin";

if (isLogin) {
    if (role === "admin") {
        console.log("Selamat datang, Admin!");
    } else {
        console.log("Selamat datang, User!");
    }
} else {
    console.log("Silakan login terlebih dahulu");
}

// Contoh validasi form
let nama = "John";
let email = "john@example.com";

if (nama) {
    if (email.includes("@")) {
        console.log("Form valid");
    } else {
        console.log("Email tidak valid");
    }
} else {
    console.log("Nama tidak boleh kosong");
}

5. Best Practices

  • Gunakan if untuk kondisi yang sederhana
  • Gunakan else if untuk kondisi yang lebih dari satu
  • Hindari nested if yang terlalu dalam
  • Gunakan else sebagai kondisi default
  • Gunakan tanda kurung untuk mengklarifikasi kondisi
  • Pisahkan kondisi kompleks ke dalam fungsi

6. Contoh Praktis

Contoh Validasi Form 
// Fungsi validasi login
function validateLogin(username, password) {
    if (!username) {
        return "Username tidak boleh kosong";
    }
    if (!password) {
        return "Password tidak boleh kosong";
    }
    if (password.length < 6) {
        return "Password minimal 6 karakter";
    }
    return true;
}

// Fungsi validasi umur
function validateAge(age) {
    if (typeof age !== 'number') {
        return "Umur harus angka";
    }
    if (age < 0) {
        return "Umur tidak boleh negatif";
    }
    if (age < 18) {
        return "Anda belum cukup umur";
    }
    return true;
}

// Contoh penggunaan
console.log(validateLogin("user", "pass123")); // true
console.log(validateLogin("", "pass123")); // "Username tidak boleh kosong"
console.log(validateAge(25)); // true
console.log(validateAge(-5)); // "Umur tidak boleh negatif"
