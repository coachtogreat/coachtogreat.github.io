Operator Perbandingan dalam JavaScript

1. Operator Dasar

Operator Arti Contoh Hasil
== Perbandingan nilai (loose equality) 5 == "5" true
=== Perbandingan nilai dan tipe (strict equality) 5 === "5" false
!= Tidak sama dengan (loose inequality) 5 != "5" false
!== Tidak sama dengan (strict inequality) 5 !== "5" true
> Lebih besar dari 10 > 5 true
< Lebih kecil dari 3 < 7 true
>= Lebih besar atau sama dengan 5 >= 5 true
<= Lebih kecil atau sama dengan 3 <= 3 true

2. Loose vs Strict Comparison

Contoh Loose vs Strict Comparison 
// Loose comparison (==)
console.log(5 == "5"); // true
console.log(0 == ""); // true
console.log(0 == false); // true

// Strict comparison (===)
console.log(5 === "5"); // false
console.log(0 === ""); // false
console.log(0 === false); // false

3. Perbandingan String

Contoh Perbandingan String 
// Perbandingan string
console.log("apple" < "banana"); // true
console.log("apple" === "Apple"); // false

// Mengubah case untuk perbandingan
console.log("apple".toLowerCase() === "Apple".toLowerCase()); // true

// Menggunakan localeCompare untuk perbandingan string
console.log("apple".localeCompare("banana")); // -1 (apple lebih kecil)
console.log("banana".localeCompare("apple")); // 1 (banana lebih besar)

4. Perbandingan dengan Boolean

Contoh Perbandingan Boolean 
// Nilai yang dianggap false
console.log(Boolean(0)); // false
console.log(Boolean("")); // false
console.log(Boolean(null)); // false
console.log(Boolean(undefined)); // false
console.log(Boolean(NaN)); // false

// Nilai yang dianggap true
console.log(Boolean(1)); // true
console.log(Boolean("text")); // true
console.log(Boolean([])); // true
console.log(Boolean({})); // true

5. Perbandingan dengan Array dan Object

Contoh Perbandingan Array dan Object 
// Perbandingan array
let arr1 = [1, 2, 3];
let arr2 = [1, 2, 3];
console.log(arr1 === arr2); // false (membandingkan referensi)

// Perbandingan object
let obj1 = { nama: "John" };
let obj2 = { nama: "John" };
console.log(obj1 === obj2); // false (membandingkan referensi)

// Membandingkan isi array/object
function arraysEqual(arr1, arr2) {
    return JSON.stringify(arr1) === JSON.stringify(arr2);
}
console.log(arraysEqual(arr1, arr2)); // true

6. Best Practices

  • Gunakan === dan !== untuk perbandingan yang lebih aman
  • Hindari penggunaan == dan != kecuali diperlukan
  • Gunakan localeCompare untuk perbandingan string yang lebih akurat
  • Gunakan JSON.stringify() untuk membandingkan isi array/object
  • Gunakan Boolean() untuk memeriksa nilai boolean

7. Contoh Praktis

Contoh Validasi Form 
// Validasi usia
function validateAge(usia) {
    if (typeof usia === 'number' && usia >= 18) {
        return "Anda sudah dewasa";
    } else if (typeof usia === 'number' && usia < 18) {
        return "Anda masih anak-anak";
    } else {
        return "Masukkan usia yang valid";
    }
}

// Validasi email
function validateEmail(email) {
    if (typeof email === 'string' && email.includes('@')) {
        return true;
    }
    return false;
}

// Contoh penggunaan
console.log(validateAge(25)); // "Anda sudah dewasa"
console.log(validateEmail("user@example.com")); // true
