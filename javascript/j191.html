Fetch API dalam JavaScript

1. Pengenalan Fetch API

Contoh Dasar 
// 1. Fetch GET sederhana
fetch('https://api.example.com/data')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Error:', error));

// 2. Fetch dengan async/await
async function fetchData() {
    try {
        const response = await fetch('https://api.example.com/data');
        const data = await response.json();
        console.log(data);
    } catch (error) {
        console.error('Error:', error);
    }
}

// 3. Fetch dengan parameter
const url = 'https://api.example.com/data';
const options = {
    method: 'GET',
    headers: {
        'Content-Type': 'application/json'
    }
};

fetch(url, options)
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Error:', error));

// 4. Fetch POST
const url = 'https://api.example.com/data';
const data = {
    nama: 'John',
    umur: 30
};

fetch(url, {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json'
    },
    body: JSON.stringify(data)
})
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error('Error:', error));

4. Best Practices

  • Gunakan async/await untuk kode yang lebih bersih
  • Tambahkan error handling
  • Gunakan try-catch untuk error handling
  • Tambahkan timeout untuk request
  • Beri nama yang deskriptif
  • Tambahkan komentar untuk menjelaskan fungsi
  • Gunakan proper error handling

5. Contoh Praktis

Contoh Penggunaan 
// 1. Fetch Data User
async function getUser(id) {
    try {
        const response = await fetch(`https://api.example.com/users/${id}`);
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Error:', error);
        throw error;
    }
}

// 2. Fetch Data dengan Timeout
function fetchWithTimeout(url, options = {}, timeout = 5000) {
    return Promise.race([
        fetch(url, options),
        new Promise((_, reject) => 
            setTimeout(() => reject(new Error('Timeout')), timeout)
        )
    ]);
}

// 3. Fetch dengan Retry
async function fetchWithRetry(url, options = {}, retries = 3) {
    for (let i = 0; i < retries; i++) {
        try {
            const response = await fetch(url, options);
            if (response.ok) {
                return response;
            }
        } catch (error) {
            if (i === retries - 1) {
                throw error;
            }
        }
    }
}

// 4. Fetch dengan Cache
async function fetchWithCache(url, options = {}) {
    const cache = localStorage.getItem(url);
    if (cache) {
        return JSON.parse(cache);
    }
    
    const response = await fetch(url, options);
    const data = await response.json();
    
    localStorage.setItem(url, JSON.stringify(data));
    return data;
}

// 5. Fetch dengan Interceptor
const fetchInterceptor = {
    request: (url, options) => {
        console.log('Request:', url);
        return [url, options];
    },
    
    response: (response) => {
        console.log('Response:', response.status);
        return response;
    }
};

async function fetchWithInterceptor(url, options = {}) {
    const [urlWithInterceptor, optionsWithInterceptor] = fetchInterceptor.request(url, options);
    const response = await fetch(urlWithInterceptor, optionsWithInterceptor);
    return fetchInterceptor.response(response);
}

// 6. Fetch dengan Middleware
async function fetchWithMiddleware(url, options = {}) {
    const middleware = [
        (url, options) => {
            options.headers = {
                ...options.headers,
                'X-Requested-With': 'XMLHttpRequest'
            };
            return [url, options];
        },
        (url, options) => {
            options.credentials = 'include';
            return [url, options];
        }
    ];

    let [finalUrl, finalOptions] = [url, options];
    for (const mw of middleware) {
        [finalUrl, finalOptions] = mw(finalUrl, finalOptions);
    }

    return fetch(finalUrl, finalOptions);
}
