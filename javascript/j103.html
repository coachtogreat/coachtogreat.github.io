Manipulasi Teks, Atribut, dan Style dalam JavaScript

1. Manipulasi Teks

Contoh Manipulasi Teks 
// HTML
<p id="paragraf">Konten awal</p>

// JavaScript
// Mengubah teks
const paragraf = document.getElementById("paragraf");
paragraf.textContent = "Konten baru";

// Menambah teks
paragraf.textContent += " tambahan";

// Mengambil teks
const teks = paragraf.textContent;
console.log(teks); // "Konten baru tambahan"

// Mengganti teks dengan HTML
paragraf.innerHTML = "<strong>Konten</strong> baru";

// Menambah HTML
paragraf.innerHTML += "<em>tambahan</em>";

2. Manipulasi Atribut

Contoh Manipulasi Atribut 
// HTML
<img id="gambar" src="gambar.jpg" alt="Gambar">

// JavaScript
// Mengambil atribut
const gambar = document.getElementById("gambar");
console.log(gambar.src); // "gambar.jpg"
console.log(gambar.getAttribute("alt")); // "Gambar"

// Mengubah atribut
gambar.src = "baru.jpg";
gambar.setAttribute("alt", "Gambar baru");

// Menghapus atribut
gambar.removeAttribute("alt");

// Mengganti atribut dengan dataset
const tombol = document.getElementById("tombol");
tombol.dataset.status = "aktif";
console.log(tombol.dataset.status); // "aktif"

3. Manipulasi Style

Contoh Manipulasi Style 
// HTML
<div id="konten">Konten</div>

// JavaScript
// Mengubah style
const konten = document.getElementById("konten");
konten.style.color = "red";
konten.style.backgroundColor = "white";
konten.style.padding = "10px";

// Mengambil style
console.log(konten.style.color); // "red"

// Menggunakan class
konten.className = "aktif";

// Mengganti multiple style
Object.assign(konten.style, {
    color: "blue",
    backgroundColor: "yellow",
    padding: "15px"
});

// Menggunakan CSS-in-JS
const style = {
    color: "green",
    backgroundColor: "pink",
    padding: "20px"
};
Object.assign(konten.style, style);

4. Best Practices

  • Gunakan className untuk mengubah style
  • Hindari penggunaan inline style jika tidak diperlukan
  • Gunakan dataset untuk menyimpan data
  • Tambahkan komentar untuk menjelaskan manipulasi
  • Pisahkan logika yang rumit ke dalam fungsi
  • Gunakan CSS-in-JS untuk style kompleks

5. Contoh Praktis

Contoh Penggunaan 
// Contoh form validation
function validateForm() {
    const form = document.getElementById("form");
    const nama = form.querySelector("#nama");
    
    if (nama.value.trim().length === 0) {
        nama.style.borderColor = "red";
        nama.style.backgroundColor = "pink";
        nama.style.padding = "5px";
    }
}

// Contoh galeri foto
function buatGaleri() {
    const konten = document.querySelector(".galeri");
    const gambar = [
        { src: "gambar1.jpg", alt: "Gambar 1" },
        { src: "gambar2.jpg", alt: "Gambar 2" }
    ];
    
    gambar.forEach(img => {
        const element = document.createElement("img");
        element.src = img.src;
        element.alt = img.alt;
        element.style.width = "200px";
        element.style.height = "200px";
        konten.appendChild(element);
    });
}

// Contoh todo list
function tambahTodo() {
    const input = document.querySelector("#todo-input");
    const list = document.querySelector("#todo-list");
    
    if (input.value.trim()) {
        const li = document.createElement("li");
        li.textContent = input.value;
        li.className = "todo-item";
        li.dataset.status = "belum";
        list.appendChild(li);
        input.value = "";
    }
}

// Contoh animasi
function animasi() {
    const kotak = document.querySelector(".kotak");
    
    let posisi = 0;
    const animasi = setInterval(() => {
        posisi++;
        kotak.style.left = posisi + "px";
        
        if (posisi >= 400) {
            clearInterval(animasi);
        }
    }, 10);
}

  • Pisahkan logika yang rumit ke dalam fungsi
  • Gunakan CSS-in-JS untuk style kompleks
