While dan do...while Loop dalam JavaScript

1. Pengenalan While Loop

Syntax While Loop 
while (kondisi) {
    // kode yang dijalankan
}

2. Contoh While Loop

Contoh While 
// Contoh hitung mundur
let i = 5;
while (i > 0) {
    console.log(i);
    i--;
}

// Contoh array
let buah = ["apel", "mangga", "jeruk"];
let index = 0;
while (index < buah.length) {
    console.log(buah[index]);
    index++;
}

// Contoh dengan kondisi kompleks
let nilai = 1;
while (nilai <= 100 && nilai % 2 === 0) {
    console.log(nilai);
    nilai += 2;
}

3. Pengenalan do...while Loop

Syntax do...while 
do {
    // kode yang dijalankan
} while (kondisi);

4. Contoh do...while Loop

Contoh do...while 
// Contoh sederhana
do {
    console.log("Hello");
} while (false);

// Contoh game tebak angka
let jawaban = Math.floor(Math.random() * 10) + 1;
let tebakan;

console.log("Tebak angka antara 1-10!");

// Menggunakan do...while untuk memastikan minimal satu kali
// tebakan dilakukan
do {
    tebakan = prompt("Masukkan tebakan Anda:");
    if (tebakan == jawaban) {
        console.log("Benar!");
    } else {
        console.log("Salah, coba lagi!");
    }
} while (tebakan != jawaban);

// Contoh validasi input
do {
    let nama = prompt("Masukkan nama:");
    if (nama.length > 0) {
        console.log("Nama valid: " + nama);
        break;
    }
    console.log("Nama tidak boleh kosong!");
} while (true);

5. Perbandingan While dan do...while

Perbandingan 
// While: memeriksa kondisi sebelum menjalankan kode
let i = 0;
while (i < 5) {
    console.log(i);
    i++;
}

// do...while: menjalankan kode minimal satu kali
let i = 5;
do {
    console.log(i);
    i++;
} while (i < 5); // kode tetap dijalankan satu kali

6. Best Practices

  • Gunakan while untuk kondisi yang tidak pasti
  • Gunakan do...while untuk kode yang harus dijalankan minimal satu kali
  • Tambahkan kondisi break untuk menghindari infinite loop
  • Pisahkan kondisi kompleks ke dalam fungsi
  • Gunakan variabel yang deskriptif
  • Tambahkan komentar untuk menjelaskan kondisi

7. Contoh Praktis

Contoh Kalkulator 
// Fungsi hitung faktorial dengan while
function faktorial(n) {
    let hasil = 1;
    let i = 1;
    while (i <= n) {
        hasil *= i;
        i++;
    }
    return hasil;
}

// Fungsi validasi input dengan do...while
function validasiInput() {
    let input;
    do {
        input = prompt("Masukkan angka antara 1-100:");
        if (input >= 1 && input <= 100) {
            return input;
        }
        console.log("Input tidak valid!");
    } while (true);
}

// Contoh game tebak angka dengan while
do {
    let jawaban = Math.floor(Math.random() * 10) + 1;
    let tebakan;
    let skor = 100;

    console.log("Tebak angka antara 1-10!");

    while (skor > 0) {
        tebakan = prompt("Masukkan tebakan Anda:");
        if (tebakan == jawaban) {
            console.log("Benar! Skor Anda: " + skor);
            break;
        }
        skor -= 10;
        console.log("Salah, coba lagi! Skor: " + skor);
    }

    if (confirm("Main lagi?") === false) {
        break;
    }
} while (true);

// Contoh validasi password dengan do...while
function validasiPassword() {
    let password;
    do {
        password = prompt("Masukkan password (minimal 8 karakter):");
        if (password.length >= 8) {
            return password;
        }
        console.log("Password minimal 8 karakter!");
    } while (true);
}
