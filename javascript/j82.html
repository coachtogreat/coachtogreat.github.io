Properti dan Metode dalam JavaScript

1. Properti dalam Objek

Contoh Properti 
// Properti sederhana
let orang = {
    nama: "John",
    umur: 25,
    alamat: "Jakarta"
};

// Properti dengan nilai berbagai tipe
let siswa = {
    nama: "John",
    umur: 25,
    aktif: true,
    nilai: [80, 85, 90],
    info: {
        kelas: "10",
        jurusan: "IPA"
    }
};

// Properti dengan simbol
let simbol = Symbol("id");
let objek = {
    [simbol]: "nilai simbol"
};

// Properti dengan computed property
let namaProp = "warna";
let mobil = {
    [namaProp]: "merah",
    ["merek_" + "mobil"]: "Toyota"
};

2. Metode dalam Objek

Contoh Metode 
// Metode sederhana
let kalkulator = {
    tambah: function(a, b) {
        return a + b;
    },
    kurang: function(a, b) {
        return a - b;
    }
};

// Metode dengan arrow function
let siswa = {
    nama: "John",
    nilai: [80, 85, 90],
    hitungRataRata: () => {
        return this.nilai.reduce((a, b) => a + b) / this.nilai.length;
    }
};

// Metode dengan computed property
let namaMetode = "hitung";
let objek = {
    [namaMetode]: function() {
        return "hasil hitung";
    }
};

// Metode statis
function Orang() {}
Orang.cetak = function() {
    console.log("Ini metode statis");
};

3. Properti dan Metode dalam Constructor

Contoh Constructor 
// Constructor dengan properti dan metode
function Orang(nama, umur) {
    this.nama = nama;
    this.umur = umur;
    
    // Properti
    this.info = {
        alamat: "Jakarta",
        hobi: ["membaca", "menulis"]
    };
    
    // Metode
    this.sapa = function() {
        console.log("Halo, nama saya " + this.nama);
    };
}

// Constructor dengan prototype
function Produk(nama, harga) {
    this.nama = nama;
    this.harga = harga;
}

Produk.prototype.diskon = function(persen) {
    return this.harga * (1 - persen);
};

// Constructor dengan class
class Mobil {
    constructor(merek, tahun) {
        this.merek = merek;
        this.tahun = tahun;
    }
    
    info() {
        return `${this.merek} ${this.tahun}`;
    }
    
    static hitungHarga(harga, diskon) {
        return harga * (1 - diskon);
    }
}

4. Best Practices

  • Gunakan arrow function untuk metode yang tidak menggunakan this
  • Gunakan prototype untuk metode yang digunakan bersama
  • Beri nama properti dan metode yang deskriptif
  • Tambahkan komentar untuk menjelaskan fungsi
  • Pisahkan logika yang rumit ke dalam fungsi
  • Hindari properti dengan nama yang sama

5. Contoh Praktis

Contoh Penggunaan 
// Contoh kalkulator
let kalkulator = {
    nilai1: 0,
    nilai2: 0,
    
    setNilai: function(a, b) {
        this.nilai1 = a;
        this.nilai2 = b;
    },
    
    tambah: function() {
        return this.nilai1 + this.nilai2;
    },
    
    kurang: function() {
        return this.nilai1 - this.nilai2;
    }
};

// Contoh data siswa
let siswa = {
    nama: "John",
    nilai: [80, 85, 90],
    
    hitungRataRata: function() {
        return this.nilai.reduce((a, b) => a + b) / this.nilai.length;
    },
    
    cetak: function() {
        return `Nama: ${this.nama}, Rata-rata: ${this.hitungRataRata()}`;
    }
};

// Contoh data produk
let produk = {
    id: 1,
    nama: "Laptop",
    harga: 10000000,
    stok: 10,
    
    hitungHargaAkhir: function(diskon) {
        return this.harga * (1 - diskon);
    },
    
    updateStok: function(jumlah) {
        this.stok += jumlah;
        return this.stok;
    }
};

// Contoh data menu
let menu = {
    makanan: {
        "Nasi Goreng": {
            harga: 20000,
            stok: 50,
            pesan: function(jumlah) {
                this.stok -= jumlah;
                return jumlah * this.harga;
            }
        },
        "Mie Goreng": {
            harga: 25000,
            stok: 40,
            pesan: function(jumlah) {
                this.stok -= jumlah;
                return jumlah * this.harga;
            }
        }
    },
    
    hitungTotal: function(makanan, jumlah) {
        return this.makanan[makanan].pesan(jumlah);
    }
};

6. Tips Penggunaan

  • Gunakan arrow function untuk metode yang tidak menggunakan this
  • Gunakan prototype untuk metode yang digunakan bersama
  • Beri nama properti dan metode yang deskriptif
  • Tambahkan komentar untuk menjelaskan fungsi
  • Pisahkan logika yang rumit ke dalam fungsi
  • Hindari properti dengan nama yang sama
