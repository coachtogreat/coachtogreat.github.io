Operator Logika dalam JavaScript

1. Operator AND (&&)

Contoh Operator AND 
// Contoh sederhana
console.log(true && true); // true
console.log(true && false); // false
console.log(false && true); // false
console.log(false && false); // false

// Dengan kondisi
let umur = 25;
let isStudent = true;

if (umur >= 18 && isStudent) {
    console.log("Anda adalah mahasiswa dewasa");
}

// Short-circuit evaluation
console.log("Hello" && "World"); // "World"
console.log(0 && "Hello"); // 0

2. Operator OR (||)

Contoh Operator OR 
// Contoh sederhana
console.log(true || true); // true
console.log(true || false); // true
console.log(false || true); // true
console.log(false || false); // false

// Dengan kondisi
let hasLogin = false;
let hasToken = true;

if (hasLogin || hasToken) {
    console.log("Anda bisa mengakses sistem");
}

// Default value
let nama = "";
let defaultName = nama || "User";
console.log(defaultName); // "User"

3. Operator NOT (!)

Contoh Operator NOT 
// Contoh sederhana
console.log(!true); // false
console.log(!false); // true

// Dengan kondisi
let isLoggedIn = false;

if (!isLoggedIn) {
    console.log("Silakan login terlebih dahulu");
}

// Membuat nilai menjadi boolean
console.log(!!"Hello"); // true
console.log(!!""); // false

4. Operator Nullish Coalescing (??)

Contoh Nullish Coalescing 
// Contoh OR biasa
let nama = "";
let defaultName = nama || "User";
console.log(defaultName); // "User" (karena string kosong dianggap false)

// Contoh Nullish Coalescing
let nama = "";
let defaultName = nama ?? "User";
console.log(defaultName); // "" (karena string kosong bukan null/undefined)

// Contoh lain
let data = null;
let nilai = data ?? "Default";
console.log(nilai); // "Default"

5. Operator Ternary (?:)

Contoh Operator Ternary 
// Contoh sederhana
let umur = 25;
let status = umur >= 18 ? "Dewasa" : "Anak-anak";
console.log(status); // "Dewasa"

// Dengan kondisi kompleks
let jumlah = 10;
let hasil = jumlah > 0 ? (jumlah < 5 ? "Kecil" : "Besar") : "Nol";
console.log(hasil); // "Besar"

// Dengan assignment
let nilai = 75;
let grade = nilai >= 70 ? "A" : nilai >= 60 ? "B" : "C";
console.log(grade); // "A"

6. Best Practices

  • Gunakan && untuk kondisi yang harus semua true
  • Gunakan || untuk kondisi yang cukup satu true
  • Gunakan ! untuk membalik nilai boolean
  • Gunakan ?? untuk default value yang lebih aman
  • Hindari penggunaan terlalu banyak operator logika
  • Gunakan tanda kurung untuk mengklarifikasi prioritas

7. Contoh Praktis

Contoh Validasi Form 
// Validasi login
function validateLogin(username, password) {
    if (username && password) {
        return true;
    }
    return false;
}

// Validasi umur
function validateAge(age) {
    return age >= 18 && age <= 60;
}

// Validasi email
function validateEmail(email) {
    return email && email.includes('@') && email.includes('.');
}

// Contoh penggunaan
console.log(validateLogin("user", "pass")); // true
console.log(validateAge(25)); // true
console.log(validateEmail("user@example.com")); // true
