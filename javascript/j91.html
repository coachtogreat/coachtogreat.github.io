Scope Global vs Lokal dalam JavaScript

1. Scope Global

Contoh Scope Global 
// Variabel global
let nama = "John";

function sapa() {
    console.log(nama); // "John"
}

// Fungsi global
function tambah(a, b) {
    return a + b;
}

// Objek global
let orang = {
    nama: "John",
    umur: 25
};

// Scope global dalam browser
console.log(window.nama); // "John"
console.log(window.tambah(5, 3)); // 8
console.log(window.orang); // {nama: "John", umur: 25}

2. Scope Lokal

Contoh Scope Lokal 
// Scope lokal dalam fungsi
function hitung() {
    let nilai = 10;
    const nama = "John";
    
    console.log(nilai); // 10
    console.log(nama); // "John"
}

hitung();
// console.log(nilai); // Error: nilai is not defined
// console.log(nama); // Error: nama is not defined

// Scope lokal dalam blok if
if (true) {
    let kondisi = "benar";
    const nilai = 100;
    
    console.log(kondisi); // "benar"
    console.log(nilai); // 100
}

// console.log(kondisi); // Error: kondisi is not defined
// console.log(nilai); // Error: nilai is not defined

// Scope lokal dalam loop
for (let i = 0; i < 5; i++) {
    console.log(i); // 0, 1, 2, 3, 4
}

// console.log(i); // Error: i is not defined

3. Perbedaan Scope

Perbedaan Scope 
// Contoh perbedaan scope
let namaGlobal = "John";

function fungsi() {
    let namaLokal = "Jane";
    
    console.log(namaGlobal); // "John"
    console.log(namaLokal); // "Jane"
}

fungsi();

console.log(namaGlobal); // "John"
// console.log(namaLokal); // Error: namaLokal is not defined

// Contoh shadowing
let nilai = 10;

function hitung() {
    let nilai = 20;
    console.log(nilai); // 20
}

hitung();
console.log(nilai); // 10

// Contoh hoisting
function contoh() {
    console.log(teks); // undefined
    var teks = "Hello";
}

contoh();

4. Best Practices

  • Hindari penggunaan variabel global yang berlebihan
  • Gunakan scope lokal untuk variabel sementara
  • Gunakan const untuk nilai yang tidak berubah
  • Beri nama variabel yang deskriptif
  • Tambahkan komentar untuk menjelaskan scope
  • Hindari shadowing variabel

5. Contoh Praktis

Contoh Penggunaan 
// Contoh kalkulator
function kalkulator() {
    let hasil = 0;
    
    function tambah(a, b) {
        hasil = a + b;
        return hasil;
    }
    
    function kurang(a, b) {
        hasil = a - b;
        return hasil;
    }
    
    return {
        tambah: tambah,
        kurang: kurang,
        getHasil: function() {
            return hasil;
        }
    };
}

const calc = kalkulator();
console.log(calc.tambah(5, 3)); // 8
console.log(calc.kurang(10, 4)); // 6

// Contoh form validation
function validateForm() {
    const form = document.getElementById("form");
    const nama = form.querySelector("#nama");
    const email = form.querySelector("#email");
    
    function validateEmail() {
        const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        return regex.test(email.value);
    }
    
    function validateNama() {
        return nama.value.trim().length > 0;
    }
    
    return validateNama() && validateEmail();
}

// Contoh game
function game() {
    let skor = 0;
    let nyawa = 3;
    
    function tambahSkor() {
        skor += 10;
        return skor;
    }
    
    function kurangNyawa() {
        nyawa--;
        return nyawa;
    }
    
    return {
        tambahSkor: tambahSkor,
        kurangNyawa: kurangNyawa,
        getSkor: function() {
            return skor;
        },
        getNyawa: function() {
            return nyawa;
        }
    };
}

const gameInstance = game();
console.log(gameInstance.tambahSkor()); // 10
console.log(gameInstance.kurangNyawa()); // 2

  • Beri nama variabel yang deskriptif
  • Tambahkan komentar untuk menjelaskan scope
