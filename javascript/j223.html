2. Nama Variabel dan Fungsi

Best Practices Nama Variabel 
// 1. Nama Variabel yang Deskriptif
// Baik:
const userCount = 10;
// Kurang Baik:
const x = 10;

// 2. Nama Fungsi yang Jelas
class User {
    // Baik:
    calculateTotalPrice(items) {
        return items.reduce((sum, item) => sum + item.price, 0);
    }
    
    // Kurang Baik:
    calc(items) {
        return items.reduce((sum, item) => sum + item.price, 0);
    }
}

// 3. Nama Konstanta
const MAX_USERS = 100;
const API_URL = 'https://api.example.com';

// 4. Nama Class
// Baik:
class UserProfile {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
}

// Kurang Baik:
class U {
    constructor(n, a) {
        this.n = n;
        this.a = a;
    }
}

3. Struktur Kode

Best Practices Struktur Kode 
// 1. Indentasi dan Spasi
function calculateTotal(items) {
    return items.reduce((sum, item) => sum + item.price, 0);
}

// 2. Panjang Fungsi
// Baik:
function calculateTotal(items) {
    return items.reduce((sum, item) => sum + item.price, 0);
}

// Kurang Baik:
function calculateTotal(items) {
    let total = 0;
    for (let i = 0; i < items.length; i++) {
        total += items[i].price;
    }
    return total;
}

// 3. Error Handling

function calculateTotal(items) {

}

4. Best Practices Coding Style

  • Gunakan nama variabel yang deskriptif
  • Ikuti konvensi penamaan
  • Gunakan comments yang bermakna
  • Ikuti standar indentasi
  • Gunakan error handling yang baik
  • Ikuti SOLID principles
  • Gunakan DRY (Don't Repeat Yourself)
  • Ikuti KISS (Keep It Simple, Stupid)

// 5. Clean Code dalam Praktek
// Baik:
function calculateUserAge(birthDate) {
    const today = new Date();
    const birth = new Date(birthDate);
    
    let age = today.getFullYear() - birth.getFullYear();
    const m = today.getMonth() - birth.getMonth();
    
    if (m < 0 || (m === 0 && today.getDate() < birth.getDate())) {
        age--;
    }
    
    return age;
}

// Kurang Baik:
function age(b) {
    const t = new Date();
    const b = new Date(b);
    let a = t.getFullYear() - b.getFullYear();
    const m = t.getMonth() - b.getMonth();
    
    if (m < 0 || (m === 0 && t.getDate() < b.getDate())) {
        a--;
    }
    
    return a;
}
