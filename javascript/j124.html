Async/Await dalam JavaScript

1. Pengenalan Async/Await

Contoh Dasar 
// Contoh dasar async/await
async function proses() {
    try {
        const hasil = await new Promise((resolve) => {
            setTimeout(() => resolve("Berhasil!"), 2000);
        });
        console.log(hasil); // "Berhasil!"
    } catch (error) {
        console.error(error);
    }
}

// Contoh dengan Promise
async function prosesData() {
    try {
        const hasil1 = await proses1();
        const hasil2 = await proses2(hasil1);
        const hasil3 = await proses3(hasil2);
        console.log(`Hasil: ${hasil3}`);
    } catch (error) {
        console.error(error);
    }
}

// Contoh dengan Promise.all
async function prosesSemua() {
    try {
        const [hasil1, hasil2] = await Promise.all([
            proses1(),
            proses2()
        ]);
        console.log(`Hasil 1: ${hasil1}, Hasil 2: ${hasil2}`);
    } catch (error) {
        console.error(error);
    }
}

2. Keuntungan Async/Await

Keuntungan 
// 1. Lebih mudah dibaca
// Callback style
proses1().then(hasil1 => {
    proses2(hasil1).then(hasil2 => {
        proses3(hasil2).then(hasil3 => {
            console.log(hasil3);
        });
    });
});

// Async/await style
async function proses() {
    const hasil1 = await proses1();
    const hasil2 = await proses2(hasil1);
    const hasil3 = await proses3(hasil2);
    console.log(hasil3);
}

// 2. Lebih mudah ditangani error
// Callback style
proses1()
    .then(hasil1 => proses2(hasil1))
    .then(hasil2 => proses3(hasil2))
    .catch(error => {
        console.error(error);
    });

// Async/await style
async function proses() {
    try {
        const hasil1 = await proses1();
        const hasil2 = await proses2(hasil1);
        const hasil3 = await proses3(hasil2);
        console.log(hasil3);
    } catch (error) {
        console.error(error);
    }
}

3. Best Practices

  • Gunakan async/await untuk operasi async
  • Tambahkan try/catch untuk error handling
  • Gunakan await untuk operasi yang memerlukan hasil sebelumnya
  • Hindari penggunaan async/await yang berlebihan
  • Beri nama fungsi yang deskriptif
  • Tambahkan komentar untuk menjelaskan async/await
  • Pisahkan logika ke dalam fungsi terpisah

4. Contoh Praktis

Contoh Penggunaan 
// Contoh todo list
async function tambahTodo(teks) {
    try {
        const li = document.createElement("li");
        li.textContent = teks;
        document.getElementById("todo-list").appendChild(li);
        
        await new Promise(resolve => setTimeout(resolve, 500));
        li.style.color = "blue";
        
        return li;
    } catch (error) {
        console.error(error);
    }
}

// Contoh galeri foto
async function tampilkanGambar(url) {
    try {
        const response = await fetch(url);
        const blob = await response.blob();
        const img = new Image();
        img.src = URL.createObjectURL(blob);
        
        await new Promise(resolve => setTimeout(resolve, 500));
        document.body.appendChild(img);
        
        return img;
    } catch (error) {
        console.error(error);
    }
}

// Contoh form validation
async function validateForm() {
    try {
        const nama = document.getElementById("nama");
        const email = document.getElementById("email");
        
        if (nama.value.trim().length === 0) {
            throw new Error("Nama tidak boleh kosong");
        }
        
        if (!email.value.includes("@")) {
            throw new Error("Email tidak valid");
        }
        
        return "Form valid";
    } catch (error) {
        console.error(error);
    }
}

// Contoh animasi
async function animasi(element) {
    try {
        let posisi = 0;
        
        while (posisi < 400) {
            element.style.left = posisi + "px";
            posisi++;
            await new Promise(resolve => setTimeout(resolve, 10));
        }
        
        return "Animasi selesai!";
    } catch (error) {
        console.error(error);
    }
}

// Contoh fetch data
async function fetchData(url) {
    try {
        const response = await fetch(url);
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        const data = await response.json();
        return data;
    } catch (error) {
        console.error(error);
    }
}

// Contoh dengan Promise.all
async function prosesSemua() {
    try {
        const [hasil1, hasil2] = await Promise.all([
            proses1(),
            proses2()
        ]);
        console.log(`Hasil 1: ${hasil1}, Hasil 2: ${hasil2}`);
    } catch (error) {
        console.error(error);
    }
}
