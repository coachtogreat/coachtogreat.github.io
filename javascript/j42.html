Switch Statement dalam JavaScript

1. Pengenalan Switch

Struktur Switch 
switch (expression) {
    case value1:
        // kode yang dijalankan jika expression === value1
        break;
    case value2:
        // kode yang dijalankan jika expression === value2
        break;
    default:
        // kode yang dijalankan jika tidak ada case yang cocok
}

2. Contoh Switch Sederhana

Contoh Switch 
// Contoh hari
let hari = "Minggu";
switch (hari) {
    case "Senin":
        console.log("Hari pertama kerja");
        break;
    case "Sabtu":
        console.log("Akhir pekan");
        break;
    case "Minggu":
        console.log("Hari libur");
        break;
    default:
        console.log("Hari kerja");
}

// Contoh angka
let nilai = 3;
switch (nilai) {
    case 1:
        console.log("Satu");
        break;
    case 2:
        console.log("Dua");
        break;
    case 3:
        console.log("Tiga");
        break;
    default:
        console.log("Lainnya");
}

3. Multiple Case

Contoh Multiple Case 
// Contoh bulan
let bulan = "April";
switch (bulan) {
    case "Januari":
    case "Maret":
    case "Mei":
    case "Juli":
    case "Agustus":
    case "Oktober":
    case "Desember":
        console.log("Bulan ini memiliki 31 hari");
        break;
    case "April":
    case "Juni":
    case "September":
    case "November":
        console.log("Bulan ini memiliki 30 hari");
        break;
    case "Februari":
        console.log("Bulan ini memiliki 28 atau 29 hari");
        break;
    default:
        console.log("Bulan tidak valid");
}

4. Fallthrough

Contoh Fallthrough 
// Contoh level user
let level = "gold";
switch (level) {
    case "bronze":
        console.log("Diskon 5%" + "\n");
    case "silver":
        console.log("Diskon 10%" + "\n");
    case "gold":
        console.log("Diskon 15%" + "\n");
    case "platinum":
        console.log("Diskon 20%" + "\n");
        break;
    default:
        console.log("Tidak ada diskon");
}

// Output untuk level "gold":
// Diskon 15%
// Diskon 20%

5. Switch dengan Expression

Contoh Expression 
// Contoh kalkulator
let operator = "+";
let a = 5;
let b = 3;

switch (operator) {
    case "+":
        console.log(a + b);
        break;
    case "-":
        console.log(a - b);
        break;
    case "*":
        console.log(a * b);
        break;
    case "/":
        console.log(a / b);
        break;
    default:
        console.log("Operator tidak valid");
}

// Contoh pengkondisian
let nilai = 85;
switch (true) {
    case nilai >= 90:
        console.log("Grade A");
        break;
    case nilai >= 80:
        console.log("Grade B");
        break;
    case nilai >= 70:
        console.log("Grade C");
        break;
    default:
        console.log("Grade D");
}

6. Best Practices

  • Gunakan break untuk setiap case kecuali ada alasan spesifik
  • Gunakan default untuk menangani kasus tidak terduga
  • Hindari terlalu banyak case dalam satu switch
  • Pisahkan kode yang kompleks ke dalam fungsi
  • Gunakan switch untuk kondisi yang memiliki banyak pilihan
  • Hati-hati dengan fallthrough

7. Contoh Praktis

Contoh Menu Program 
// Fungsi menu program
function prosesMenu(pilihan) {
    switch (pilihan) {
        case "1":
            console.log("Menampilkan data");
            break;
        case "2":
            console.log("Menambah data");
            break;
        case "3":
            console.log("Menghapus data");
            break;
        case "4":
            console.log("Keluar program");
            return;
        default:
            console.log("Pilihan tidak valid");
    }
}

// Fungsi validasi input
function validateInput(input) {
    switch (typeof input) {
        case "string":
            return input.length > 0;
        case "number":
            return !isNaN(input);
        case "boolean":
            return true;
        default:
            return false;
    }
}

// Contoh penggunaan
prosesMenu("2"); // Menambah data
console.log(validateInput("Hello")); // true
