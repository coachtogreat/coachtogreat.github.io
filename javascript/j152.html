Export dan Import dalam JavaScript

1. Pengenalan Export dan Import

Contoh Dasar 
// File: todo.js
export const todos = [];

export function tambahTodo(teks) {
    todos.push(teks);
}

export function hapusTodo(id) {
    todos.splice(id, 1);
}

// File: index.js
import { todos, tambahTodo, hapusTodo } from './todo.js';

// Atau menggunakan default export
export default {
    todos,
    tambahTodo,
    hapusTodo
};

// File: index.js
import todo from './todo.js';

// Named export
export const nama = "John";
export const umur = 30;

// Default export
export default function() {
    console.log("Halo!");
}

// Multiple export
export const nama = "John";
export const umur = 30;
export function sapa() {
    console.log("Halo!");
}

2. Jenis Export

Jenis Export 
// 1. Named Export
export const nama = "John";
export const umur = 30;

// 2. Default Export
export default function() {
    console.log("Halo!");
}

// 3. Multiple Export
export const nama = "John";
export const umur = 30;
export function sapa() {
    console.log("Halo!");
}

// 4. Export dari file lain
export { nama, umur } from './data.js';

// 5. Export dengan rename
export { nama as namaUser, umur as usia } from './data.js';

// 6. Export default dan named
export const nama = "John";
export default function() {
    console.log("Halo!");
}

3. Jenis Import

Jenis Import 
// 1. Named Import
import { nama, umur } from './data.js';

// 2. Import dengan rename
import { nama as namaUser, umur as usia } from './data.js';

// 3. Default Import
import sapa from './fungsi.js';

// 4. Import semua
import * as data from './data.js';

// 5. Import default dan named
import sapa, { nama, umur } from './fungsi.js';

// 6. Import file CSS
import './style.css';

// 7. Import dynamic
const module = await import('./data.js');

// 8. Import kondisional
if (condition) {
    const module = await import('./data.js');
}

4. Best Practices

  • Gunakan default export untuk satu fungsi utama
  • Gunakan named export untuk fungsi pendukung
  • Beri nama yang deskriptif untuk export
  • Tambahkan komentar untuk menjelaskan export
  • Hindari circular dependencies
  • Gunakan import dynamic untuk lazy loading
  • Tambahkan error handling untuk import dynamic

5. Contoh Praktis

Contoh Penggunaan 
// 1. Todo List
// File: todo.js
export const todos = [];

export function tambahTodo(teks) {
    todos.push(teks);
}

export function hapusTodo(id) {
    todos.splice(id, 1);
}

// File: index.js
import { todos, tambahTodo, hapusTodo } from './todo.js';

// 2. Galeri Foto
// File: galeri.js
export const gambar = [];

export function tambahGambar(url) {
    gambar.push(url);
}

export function hapusGambar(id) {
    gambar.splice(id, 1);
}

// File: index.js
import { gambar, tambahGambar, hapusGambar } from './galeri.js';

// 3. Form Validation
// File: validator.js
export function validateEmail(email) {
    const regex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return regex.test(email);
}

export function validatePassword(password) {
    return password.length >= 8;
}

// File: index.js
import { validateEmail, validatePassword } from './validator.js';

// 4. Animasi
// File: animasi.js
export function gerakKiri(element, jarak) {
    element.style.left = jarak + "px";
}

export function gerakKanan(element, jarak) {
    element.style.right = jarak + "px";
}

// File: index.js
import { gerakKiri, gerakKanan } from './animasi.js';

// 5. Data Storage
// File: storage.js
export function simpan(key, value) {
    localStorage.setItem(key, JSON.stringify(value));
}

export function ambil(key) {
    const data = localStorage.getItem(key);
    return data ? JSON.parse(data) : null;
}

// File: index.js
import { simpan, ambil } from './storage.js';
