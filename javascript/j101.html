DOM Tree dalam JavaScript

1. Pengenalan DOM Tree

Struktur DOM Tree 
// Contoh struktur HTML
<html>
    <head>
        <title>Judul Halaman</title>
    </head>
    <body>
        <div class="konten">
            <p>Paragraf pertama</p>
            <p>Paragraf kedua</p>
        </div>
    </body>
</html>

// Struktur DOM Tree
html
├── head
│   └── title
└── body
    └── div.konten
        ├── p
        └── p

2. Jenis Node dalam DOM

Jenis Node 
// Element Node
<div class="konten"></div>

// Text Node
"Paragraf pertama"

// Attribute Node
"class="konten""

// Comment Node
<!-- Ini adalah komentar -->

// Document Node
<!DOCTYPE html>
<html>...</html>

3. Properti DOM Tree

Contoh Properti 
// Mendapatkan elemen
const div = document.querySelector("div");

// Properti node
console.log(div.nodeType); // 1 (element node)
console.log(div.nodeName); // "DIV"
console.log(div.nodeValue); // null

// Properti element
console.log(div.className); // "konten"
console.log(div.id); // "id"
console.log(div.textContent); // "Paragraf pertama"

// Properti parent
console.log(div.parentNode); // element parent
console.log(div.parentElement); // element parent

// Properti sibling
console.log(div.previousSibling); // node sebelumnya
console.log(div.nextSibling); // node selanjutnya

// Properti child
console.log(div.firstChild); // node pertama
console.log(div.lastChild); // node terakhir
console.log(div.childNodes); // NodeList semua child

4. Metode DOM Tree

Contoh Metode 
// Membuat node baru
const div = document.createElement("div");
const text = document.createTextNode("Hello");

// Menambahkan node
const parent = document.querySelector("#parent");
parent.appendChild(div);
parent.insertBefore(div, parent.firstChild);

// Menghapus node
const child = document.querySelector("#child");
child.remove();
parent.removeChild(child);

// Mengganti node
const baru = document.createElement("p");
const lama = document.querySelector("#lama");
parent.replaceChild(baru, lama);

// Menyalin node
const asli = document.querySelector("#asli");
const salinan = asli.cloneNode(true);

5. Best Practices

  • Gunakan querySelector untuk seleksi element
  • Hindari penggunaan innerHTML jika tidak diperlukan
  • Gunakan createElement untuk membuat element baru
  • Tambahkan komentar untuk menjelaskan manipulasi DOM
  • Pisahkan logika yang rumit ke dalam fungsi
  • Gunakan event delegation untuk event handling

6. Contoh Praktis

Contoh Penggunaan 
// Contoh membuat list
function tambahList() {
    const ul = document.querySelector("#list");
    const li = document.createElement("li");
    li.textContent = "Item baru";
    ul.appendChild(li);
}

// Contoh menghapus item
function hapusItem(event) {
    const li = event.target;
    li.remove();
}

// Contoh mengganti konten
function ubahKonten() {
    const p = document.querySelector("#paragraf");
    p.textContent = "Konten baru";
}

// Contoh menyalin element
function salinElement() {
    const asli = document.querySelector("#asli");
    const salinan = asli.cloneNode(true);
    document.body.appendChild(salinan);
}

// Contoh event delegation
function tambahEvent() {
    const parent = document.querySelector("#parent");
    parent.addEventListener("click", function(event) {
        if (event.target.classList.contains("item")) {
            event.target.remove();
        }
    });
}
