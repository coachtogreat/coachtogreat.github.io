JSON parse & stringify dalam JavaScript

1. Pengenalan JSON

Contoh Dasar 
// JSON string
const json = '{"nama": "John", "umur": 30}';

// Parsing JSON ke object
const obj = JSON.parse(json);
console.log(obj.nama); // "John"

// Stringify object ke JSON
const obj = { nama: "John", umur: 30 };
const json = JSON.stringify(obj);
console.log(json); // "{"nama":"John","umur":30}"

// Parsing dengan reviver
const json = '{"nama": "John", "umur": 30}';
const obj = JSON.parse(json, (key, value) => {
    if (key === "umur") {
        return value * 2;
    }
    return value;
});
console.log(obj.umur); // 60

// Stringify dengan replacer
const obj = { nama: "John", umur: 30, alamat: null };
const json = JSON.stringify(obj, (key, value) => {
    if (value === null) {
        return undefined;
    }
    return value;
});
console.log(json); // "{"nama":"John","umur":30}"

// Stringify dengan space
const obj = { nama: "John", umur: 30 };
const json = JSON.stringify(obj, null, 2);
console.log(json);
/*
{
  "nama": "John",
  "umur": 30
}
*/

2. Parsing JSON

Contoh Parsing 
// Parsing JSON sederhana
const json = '{"nama": "John", "umur": 30}';
const obj = JSON.parse(json);
console.log(obj.nama); // "John"

// Parsing array
const json = '[{"nama": "John"}, {"nama": "Jane"}]';
const array = JSON.parse(json);
console.log(array[0].nama); // "John"

// Parsing dengan reviver
const json = '{"nama": "John", "tanggal": "2025-01-01"}';
const obj = JSON.parse(json, (key, value) => {
    if (key === "tanggal") {
        return new Date(value);
    }
    return value;
});
console.log(obj.tanggal instanceof Date); // true

// Parsing dengan error handling
const json = '{"nama": "John", "umur": 30}';
try {
    const obj = JSON.parse(json);
    console.log(obj);
} catch (error) {
    console.error("Error parsing JSON:", error);
}

3. Stringify JSON

Contoh Stringify 
// Stringify object
const obj = { nama: "John", umur: 30 };
const json = JSON.stringify(obj);
console.log(json); // "{"nama":"John","umur":30}"

// Stringify dengan replacer
const obj = { nama: "John", umur: 30, alamat: null };
const json = JSON.stringify(obj, (key, value) => {
    if (value === null) {
        return undefined;
    }
    return value;
});
console.log(json); // "{"nama":"John","umur":30}"

// Stringify dengan space
const obj = { nama: "John", umur: 30 };
const json = JSON.stringify(obj, null, 2);
console.log(json);
/*
{
  "nama": "John",
  "umur": 30
}
*/

// Stringify dengan fungsi replacer
const obj = { nama: "John", umur: 30 };
const json = JSON.stringify(obj, (key, value) => {
    if (key === "umur") {
        return value * 2;
    }
    return value;
});
console.log(json); // "{"nama":"John","umur":60}"

// Stringify dengan array replacer
const obj = { nama: "John", umur: 30, alamat: "Jl. Contoh" };
const json = JSON.stringify(obj, ["nama", "umur"]);
console.log(json); // "{"nama":"John","umur":30}"

// Stringify dengan fungsi replacer dan space
const obj = { nama: "John", umur: 30 };
const json = JSON.stringify(obj, (key, value) => {
    if (key === "umur") {
        return value * 2;
    }
    return value;
}, 2);
console.log(json);
/*
{
  "nama": "John",
  "umur": 60
}
*/

4. Best Practices

  • Tambahkan error handling saat parsing
  • Gunakan replacer untuk mengubah nilai saat stringify
  • Gunakan reviver untuk mengubah nilai saat parsing
  • Tambahkan space untuk JSON yang lebih mudah dibaca
  • Pisahkan logika ke dalam fungsi terpisah
  • Tambahkan validasi data
  • Beri nama variabel yang deskriptif
