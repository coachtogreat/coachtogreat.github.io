FormData dalam JavaScript

1. Pengenalan FormData

Contoh Dasar 
// 1. Membuat FormData dari form
const form = document.querySelector('form');
const formData = new FormData(form);

// 2. Membuat FormData manual
const formData = new FormData();
formData.append('nama', 'John');
formData.append('umur', '30');

// 3. Membaca FormData
for (const pair of formData.entries()) {
    console.log(pair[0] + ': ' + pair[1]);
}

// 4. Mengirim FormData dengan fetch
fetch('https://api.example.com/data', {
    method: 'POST',
    body: formData
});

// 5. Menghapus data
formData.delete('nama');

// 6. Mengubah data
formData.set('umur', '35');

// 7. Mendapatkan data
const nama = formData.get('nama');
const semuaNama = formData.getAll('nama');

2. File Upload

Contoh File Upload 
// 1. Upload file sederhana
const fileInput = document.querySelector('input[type="file"]');
const formData = new FormData();
formData.append('file', fileInput.files[0]);

// 2. Upload multiple file
const fileInput = document.querySelector('input[type="file"][multiple]');
const formData = new FormData();
for (const file of fileInput.files) {
    formData.append('files', file);
}

// 3. Upload file dengan metadata
const fileInput = document.querySelector('input[type="file"]');
const formData = new FormData();
formData.append('file', fileInput.files[0], 'nama-file.jpg');
formData.append('nama', 'John');

// 4. Progress upload
const formData = new FormData();
const fileInput = document.querySelector('input[type="file"]');
formData.append('file', fileInput.files[0]);

fetch('https://api.example.com/upload', {
    method: 'POST',
    body: formData
})
.then(response => response.json())
.then(data => console.log(data))
.catch(error => console.error('Error:', error));

// 5. Upload dengan progress bar
const formData = new FormData();
const fileInput = document.querySelector('input[type="file"]');
formData.append('file', fileInput.files[0]);

const progressElement = document.querySelector('.progress');

fetch('https://api.example.com/upload', {
    method: 'POST',
    body: formData,
    onUploadProgress: progressEvent => {
        const percent = (progressEvent.loaded / progressEvent.total) * 100;
        progressElement.style.width = percent + '%';
    }
})
.then(response => response.json())
.then(data => console.log(data))
.catch(error => console.error('Error:', error));

3. Best Practices

  • Gunakan FormData untuk form dengan file upload
  • Tambahkan error handling
  • Gunakan progress bar untuk upload besar
  • Beri nama yang deskriptif
  • Tambahkan komentar untuk menjelaskan fungsi
  • Gunakan FormData untuk data binary
  • Gunakan FormData untuk multipart form data
